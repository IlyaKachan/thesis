\section{Асимптотические разложения в окрестности нуля}

В данном разделе будем придерживаться следующих компактных записей:
\begin{gather} \label{bounds_3_39}
\Delta^k[0,t] = \{(t_1,\ldots, t_k)\in [0,1]^k:\; 0\le t_1 < \ldots < t_k\le t\} \\ \nonumber
\int_{\Delta^k[0,t]} dB^{(I_k)}= \int_0^t\int_0^{t_k}\ldots 
\int_0^{t_2} dB^{(i_1)}_{t_1} \ldots dB^{(i_{k-1})}_{t_{k-1}}dB^{(i_k)}_{t_k},\\
 I_k=(i_1,\ldots, i_k)\in \mathbb{N}^k_d:=\{0,\ldots, d\}^k,\\ 
 \label{bounds_3_41}
\quad D^{(i)}_f = \sum_{j=1}^n f_{ji}(x)\frac{\partial}{\partial x_j},\; 
i \in \{0,\ldots, d\}\qquad D^{(I_k)}_f = D^{(i_1)}_f\ldots D^{(i_k)}_f,\\
\textbf{P}_t g(x) = \E g(X^{x}_t),\quad t\ge 0.
\end{gather}
В дальнейшем для краткости будем опускать верхний индекс $x$ и обозначать 
решение через $X_t$ в доказательствах.

\begin{theorem}\label{chapter_3_asympt_development}
Пусть $f\in C^{N+2}_b(\mathbb{R}^n, \mathbb{R}^{n \times (d+1)})$,
$g\in C^{N+3}_b(\mathbb{R}^n, \mathbb{R})$, $N\in\mathbb{N}$. Тогда для любого 
фиксированного $H\in(1/3, 1/2]$ такого, что $H < H_{min} =\linebreak= 
\min\limits_{i=0,\ldots d} H_i$ справедливо следующее асимптотическое разложение:
\begin{equation}\label{bounds_3_43}
\textup{\textbf{P}}_t g(x) = g(x) + 
\sum_{k=1}^N \sum_{I_{k}\in\{0,\ldots, d\}^{k}} t^{|H_{I_k}|}\cdot 
(D^{(I_{k})}_f g)(x)\E\biggl(\int_{\Delta^{k}[0,1]}dB^{(I_{k})}\biggr) + 
O(t^{(N+1)H}),
\end{equation}
при $t\to 0$, где $|H_{I_k}| = H_{i_1} + H_{i_2}+ \ldots + H_{i_k}$ --- сумма 
индексов Харста дробных броуновских движений $B^{(i_1)}, B^{(i_2)},\ldots, 
B^{(i_k)}$.
\end{theorem}

\proof C учетом обозначений (\ref{bounds_3_41}) формулу замены переменных 
(\ref{eq_ito_formula}) можно записать в следующей развернутой форме:
\begin{equation}\label{eq_wide_ito_formula}
g(X^x_t) = g(x) + \sum_{i=0}^d \int_0^t (D^{(i)}_f g)(X^x_r)dB^{(i)}_r
\end{equation}
Применяя формулу (\ref{eq_wide_ito_formula}) $(N+1)$ раз, c учетом обозначений 
(\ref{bounds_3_39}) --- (\ref{bounds_3_41}) получим:
\begin{gather} \nonumber
g(X_t) = g(x) + \sum_{k=1}^{N} \sum_{I_k\in\mathbb{N}_d^k} (D^{(I_k)}_f g)(x) 
\int_{\Delta^k[0,t]} dB^{(I_k)} + \\ \label{bounds_3_45}
+ \sum_{I_{N+1}\in \mathbb{N}_d^{N+1}} \int_0^t\int_0^{t_{N+1}}\ldots 
\int_0^{t_2} (D^{(I_{N+1})}_f g)(X_{t_1})\, dB^{(i_1)}_{t_1} \ldots 
dB^{(i_N)}_{t_N}dB^{(i_{N+1})}_{t_{N+1}}.
\end{gather}

Обозначим $\varphi_{I_{N+1}}(x) = (D^{(I_{N+1})}_f g)(x)$ и преобразуем 
последнее слагаемое в (\ref{bounds_3_45}). Введем в рассмотрение процесс 
$\widehat B^{(c)}_u = (\widehat B^{(0; c)}_u, \widehat B^{(1; c)}_u, \ldots, 
\widehat B^{(d; c)}_u)^\top$, зависящий от параметра $c > 0$, $i$-я компонента 
которого определяется равенством $\widehat B^{(i; c)}_u = c^{H_i} B^{(i)}_{u/c}$, 
$u\in [0, T]$. По свойству самоподобия дробного броуновского движения процесс 
$\widehat B^{(i; c)}_u$ также является дробным броуновским движением с 
индексом Харста $H_i$ для любого $c>0$, $i=\overline{1,\,d}$. 
Следовательно, при фиксированном $t\in [0,T]$:
\begin{gather} \nonumber
\int_0^t\int_0^{t_{N+1}}\ldots 
\int_0^{t_2} \varphi_{I_{N+1}}(X_{t_1})\, 
dB^{(i_1)}_{t_1} \ldots dB^{(i_N)}_{t_N}dB^{(i_{N+1})}_{t_{N+1}}= \nonumber \\ 
= \int_0^1 dB^{(i_{N+1})}_{t\cdot t_{N+1}} 
\int_0^{t_{N+1}} dB^{(i_N)}_{t\cdot t_N}\ldots 
\int_0^{t_2} \varphi_{I_{N+1}}(X_{t\cdot t_1})\, dB^{(i_1)}_{t\cdot t_1}= \nonumber \\ 
\stackrel{\mathcal{L}}{=} \int_0^1 d\widehat B^{(i_{N+1};t)}_{t\cdot t_{N+1}} 
\int_0^{t_{N+1}} d\widehat B^{(i_N;t)}_{t\cdot t_N}\ldots 
\int_0^{t_2} \varphi_{I_{N+1}}(\widehat X^{(t)}_{t\cdot t_1})\, 
d\widehat B^{(i_1;t)}_{t\cdot t_1} = \nonumber \\ \label{bounds_3_46}
= t^{H_{i_1} + \ldots + H_{i_{N+1}}}\int_0^1 dB^{(i_{N+1})}_{t_{N+1}} 
\int_0^{t_{N+1}} dB^{(i_N)}_{t_N}\ldots 
\int_0^{t_2} \varphi_{I_{N+1}}(\widehat X^{(t)}_{t\cdot t_1})\, dB^{(i_1)}_{t_1},
\end{gather} 
где знак $\stackrel{\mathcal{L}}{=}$ означает совпадение распределений, а 
$\widehat X^{(t)}_\tau$ --- решение уравнения:
\begin{equation}\label{bounds_3_47}
d\widehat X^{(t)}_\tau = f(\widehat X^{(t)}_\tau)d\widehat B^{(t)}_\tau,
\quad \tau \in [0, T]
\end{equation}
с начальным условием $\widehat X^{(t)}_0 = x$. По тем же соображениям
\begin{equation}\label{bounds_3_48}
\int_{\Delta^k[0,t]} dB^{(I_k)} \,\stackrel{\mathcal{L}}{=}\; 
t^{|H_{I_k}|}\int_{\Delta^k[0,1]} dB^{(I_k)},
\end{equation}
а посему из (\ref{bounds_3_45}) --- (\ref{bounds_3_48}) после взятия 
математического ожидания, получим:
\begin{equation} \label{bounds_3_49}
\textbf{P}_t g(x) = g(x) + \sum_{k=1}^{N} \sum_{I_k\in\mathbb{N}_d^k} 
t^{|H_{I_k}|} (D^{(I_k)}_f g)(x) \E\biggl(\int_{\Delta^k[0,1]} dB^{(I_k)}\biggr)+ 
\mathcal{R}_{N+1}(t),
\end{equation}
где
\begin{gather}\nonumber
\mathcal{R}_{N+1}(t) =\\ \label{bounds_3_50}
= \!\!\!\!\!\sum_{\substack{I_{N+1}\in \mathbb{N}_d^{N+1}}} 
\Biggl( t^{|H_{I_{N+1}}|} \E\int_0^1\int_0^{t_{N+1}}\!\!\!\ldots 
\int_0^{t_2} (D^{(I_{N+1})}_f g)(\widehat X^{(t)}_{t\cdot t_1})\, 
dB^{(i_1)}_{t_1} \ldots dB^{(i_N)}_{t_N}dB^{(i_{N+1})}_{t_{N+1}}\Biggr).
\end{gather}
Поскольку $|H_{I_{N+1}}|\ge (N+1)H$ для любого $I_{N+1}$, то при $t < 1$:
\begin{gather}\nonumber
|\mathcal{R}_{N+1}(t)| \le (d+1)^{N+1} t^{(N+1)H}\times \\ \label{bounds_3_51}
\times \max_{\substack{I_{N+1}\in\mathbb{N}_d^{N+1}}} 
\E\biggl|\int_0^1\int_0^{t_{N+1}}\!\!\!\ldots \int_0^{t_2} 
(D^{(I_{N+1})}_f g)(\widehat X^{(t)}_{t\cdot t_1})\, dB^{(i_1)}_{t_1} \ldots 
dB^{(i_N)}_{t_N}dB^{(i_{N+1})}_{t_{N+1}}\biggr|.
\end{gather}
Поэтому, учитывая (\ref{bounds_3_49}) --- (\ref{bounds_3_51}), для завершения 
доказательства формулы (\ref{bounds_3_43}) осталось показать, что
\begin{equation}\label{bounds_3_52}
\E\biggl|\int_0^1\int_0^{t_{N+1}}\!\!\!\ldots \int_0^{t_2} (D^{(I_{N+1})}_f g)
(\widehat X^{(t)}_{t\cdot t_1})\, dB^{(i_1)}_{t_1} \ldots 
dB^{(i_N)}_{t_N}dB^{(i_{N+1})}_{t_{N+1}}\biggr| < +\infty
\end{equation}
для любых $I_{N+1} = (i_1, \ldots, i_{N+1})\in\mathbb{N}_d^{N+1}$.

Рассмотрим повторные интегралы чуть более общего вида, нежели (\ref{bounds_3_52}). 
Для произвольного фиксированного $c\in (0, 1]$ будем оценивать повторные интегралы 
вида
\begin{equation}
\mathcal{I}^{(k)}_t = \int_0^t\int_0^{t_{k-1}}\ldots
\int_0^{t_1}\varphi\bigl(
\widehat X^{(c)}_{c r}\bigr)\,dB^{(i_1)}_{r}\ldots 
dB^{(i_{k-2})}_{t_{k-2}}dB^{(i_{k-1})}_{t_{k-1}},\quad t\in[0,1],
\end{equation}
в которых $\varphi: \mathbb{R}^n \to \mathbb{R}$ --- произвольная функция с 
непрерывными и ограниченными производными до второго порядка включительно. 
Для единообразия положим $\mathcal{I}^{(0)}_t = 
\varphi\bigl(\widehat X^{(c)}_{c t}\bigr)$.

\begin{lemma}
Пусть $\varphi$ имеет непрерывные и ограниченные производные до второго порядка 
включительно. Тогда справедливы неравенства\linebreak 
$|\mathcal{I}^{(k)}_{s,t} - \mathcal{I}^{(k-1)}_s B^{(i)}_{s,t}|\le 
M_k |t-s|^{2H}$ для любого $i=\overline{0,d}$ и $|\mathcal{I}^{(k)}_{s,t}|\le 
\widetilde{M}_k|t-s|^H$, где $M_k, \widetilde{M}_k$~--- случайные величины 
\textup{(}не зависящие от $s, t$\textup{)}.
\end{lemma}

\proof Проведем индукцией по $k$.

Рассмотрим $k=1$. Докажем, что $\bigl(\varphi\bigl(
\widehat X^{(c)}_{c \,\cdot}\bigr)\bigr)' = 
c^{H_{i_1}} D\varphi\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)
\bigl(\widehat X^{(c)}\,\!\bigr)'_{c\,\cdot}$. Введем обозначение 
$R^{\widehat X^{(c)}; i_1}_{u,v} = \widehat X^{(c)}_{u,v} - 
\bigl(\widehat X^{(c)}\bigr)'_u \widehat B^{(i_1; c)}_{u,v}$ для $i_1$-й 
компоненты остатка $R^{\widehat X^{(c)}}$ по отношению к процессу $\widehat X^{(c)}$, 
управляемому процессом $\widehat B^{(i_1; c)}$.

Используя формулу Тейлора, будем иметь
\begin{gather}\nonumber
R^{\varphi(\widehat X^{(c)}_{c\,\cdot}); i_1}_{s,t} := 
\varphi\bigl(\widehat X^{(c)}_{c t}\bigr) - 
\varphi\bigl(\widehat X^{(c)}_{c s}\bigr) - 
c^{H_{i_1}} D\varphi\bigl(\widehat X^{(c)}_{c s}\bigr)
\bigl(\widehat X^{(c)}\bigr)'_{c s}B^{(i_1)}_{s,t}=\\ \nonumber
= \varphi\bigl(\widehat X^{(c)}_{c t}\bigr) - 
\varphi\bigl(\widehat X^{(c)}_{c s}\bigr) - 
D\varphi\bigl(\widehat X^{(c)}_{cs}\bigr)
\bigl(\widehat X^{(c)}\,\!\bigr)'_{c s}\widehat B^{(i_1; c)}_{c s, c t} =\\ 
\nonumber
= \varphi\bigl(\widehat X^{(c)}_{c t}\bigr) - 
\varphi\bigl(\widehat X^{(c)}_{c s}\bigr) - 
D\varphi\bigl(\widehat X^{(c)}_{cs}\bigr)\widehat X^{(c)}_{cs, ct} + 
D\varphi\bigl(\widehat X^{(c)}_{cs}\bigr) R^{\widehat X^{(c)}; i_1}_{cs,ct}=\\ 
\label{bounds_3_56}
= \frac{1}{2}D^2\varphi\bigl(\widehat X^{(c)}_{cs} + 
\theta \widehat X^{(c)}_{cs,ct}\bigr) \widehat X^{(c)}_{cs,ct}\otimes 
\widehat X^{(c)}_{cs,ct} + D\varphi\bigl(\widehat X^{(c)}_{cs}\bigr) 
R^{\widehat X^{(c)}; i_1}_{cs,ct}
\end{gather}
для некоторого $\theta\in(0,1)$. Из теоремы существования 
\ref{chapter_3_existance_th} следует, что 
$\|R^{\widehat X^{(c); i_1}}\|_{2H} \le \|R^{\widehat X^{(c)}}\|_{2H} < \infty$.
Легко видеть, что
\begin{gather}\nonumber
\left|R^{\varphi(\widehat X^{(c)}_{c\,\cdot}); i_1}_{s,t}\right|\le 
\frac{1}{2}\|D^2\varphi\|_\infty\|\widehat X^{(c)}\|^2_H|ct - cs|^{2H} + 
\|D\varphi\|_\infty\|R^{\widehat X^{(c); i_1}}\|_{2H}|ct-cs|^{2H} =\\ 
= c^{2H}\left(\frac{1}{2}\|D^2\varphi\|_{\infty}\|\widehat X^{(c)}\|^2_H + 
\|D\varphi\|_{\infty}\|R^{\widehat X^{(c)}}\|_{2H}\right)|t-s|^{2H},
\end{gather}
откуда следует, что $\|R^{\varphi(\widehat X^{(c)}_{c\,\cdot}); i_1}_{s,t}\|_{2H} 
\le \frac{1}{2}\|D^2\varphi\|_{\infty}\|\widehat X^{(c)}\|^2_H + 
\|D\varphi\|_{\infty}\|R^{\widehat X^{(c)}}\|_{2H} < \infty$ (так как $c\le 1$), 
что и требовалось.

Далее, поскольку $\varphi\bigl(\widehat X^{(c)}_{c \,\cdot}\bigr)' = 
c^{H_{i_1}} D\varphi\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)
\bigl(\widehat X^{(c)}\,\!\bigr)'_{c\,\cdot}=  c^{H_{i_1}} 
D\varphi\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)
f\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)$, то
\begin{gather}\nonumber
\Bigl|\Bigl(\varphi\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)'\Bigl)_{s,t}\Bigl| = 
c^{H_{i_1}}\bigl|(D\varphi\cdot f)\bigl(\widehat X^{(c)}_{ct}\bigr) - 
(D\varphi\cdot f)\bigl(\widehat X^{(c)}_{cs}\bigr)\bigr| =\\ \nonumber
= c^{H_{i_1}} \left|D(D\varphi\cdot f)\bigl(\widehat X^{(c)}_{cs} + 
\theta \widehat X^{(c)}_{cs,ct}\bigr)\right|\cdot
\bigl|\widehat X^{(c)}_{cs,ct}\bigr| =\\ \nonumber
= c^{H_{i_1}}\left|(D^2\varphi\cdot f)\bigl(\widehat X^{(c)}_{cs} + 
\theta \widehat X^{(c)}_{cs,ct}\bigr) + 
(D\varphi\cdot Df)\bigl(\widehat X^{(c)}_{cs} + 
\theta \widehat X^{(c)}_{cs,ct}\bigr)\right|\cdot
\bigl|\widehat X^{(c)}_{cs,ct}\bigr|\le \\ \nonumber
\le c^{H_{i_1}}\left(\|D^2\varphi\|_{\infty}\|f\|_\infty + 
\|D\varphi\|_{\infty}\|Df\|_{\infty}\right)\|\widehat X^{(c)}\|_H|ct-cs|^H =\\
= c^{H_{i_1}+H}\left(\|D^2\varphi\|_{\infty}\|f\|_\infty + 
\|D\varphi\|_{\infty}\|Df\|_{\infty}\right)\|\widehat X^{(c)}\|_H|t-s|^H.
\end{gather}
Отсюда следует, что 
$\bigl\|\varphi\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)'\bigr\|_H \le 
\left(\|D^2\varphi\|_{\infty}\|f\|_\infty + 
\|D\varphi\|_{\infty}\|Df\|_{\infty}\right)\|\widehat X^{(c)}\|_H <\linebreak 
<\infty$. Таким образом, ввиду неравенства $\bigl||a|-|b|\bigr|\le |a-b|$ и 
предложения~\ref{hairer_th_4_10}, будем иметь:
\begin{gather*}
\nonumber
\left|\int_s^t \varphi\bigl(\widehat X^{(c)}_{c r}\bigr)\,dB^{(i_1)}_r - 
\varphi\bigl(\widehat X^{(c)}_{c s}\bigr)B^{(i_1)}_{s,t}\right| \le 
c^{H_{i_1}}\|D\varphi\|_{\infty}\|f\|_{\infty}\|\mathbb{B}^{(i_1)}\|_{2H}
|t-s|^{2H} + \\
\nonumber
+ C \left(\|B^{(i_1)}\|_{H}
\bigl\|R^{\varphi(\widehat X^{(c)}_{c\,\cdot}); i_1}\bigr\|_{2H} + 
\|\mathbb{B}^{(i_1)}\|_{2H}\bigl\|
\varphi\bigl(\widehat X^{(c)}_{c\,\cdot}\bigr)'\bigr\|_H\right)|t-s|^{3H} \le \\
\nonumber
\le \|D\varphi\|_{\infty}\|f\|_{\infty}\|\mathbb{B}\|_{2H}|t-s|^{2H} +
\end{gather*}
\begin{gather}  
\nonumber
+ C \biggl(\left(\frac{1}{2}\|D^2\varphi\|_{\infty}\|\widehat X^{(c)}\|^2_H + 
\|D\varphi\|_{\infty}\|R^{\widehat X^{(c)}}\|_{2H}\right)\|B\|_{H} + \\ 
+ \left(\|D^2\varphi\|_{\infty}\|f\|_\infty + 
\|D\varphi\|_{\infty}\|Df\|_{\infty}\right) 
\|\widehat X^{(c)}\|_H \|\mathbb{B}\|_{2H} \biggr)|t-s|^{2H} \le M_1|t-s|^{2H}
\end{gather}
где $M_1 = \Bigl(C(\|D^2\varphi\|_{\infty}\|f\|_\infty + 
\|D\varphi\|_{\infty}\|Df\|_{\infty})\|\widehat X^{(c)}\|_H + 
\|D\varphi\|_{\infty}\|f\|_{\infty}\Bigr)\|\mathbb{B}\|_{2H} +\linebreak
+ C\Bigl(\frac{1}{2}\|D^2\varphi\|_{\infty}\|\widehat X^{(c)}\|^2_H + 
\|D\varphi\|_{\infty}\|R^{\widehat X^{(c)}}\|_{2H}\Bigr)\|B\|_{H}.$

Итак, получили 
$|\mathcal{I}^{(1)}_{s,t} - \mathcal{I}^{(0)}_s B^{(i_1)}_{s,t}|\le 
M_1 |t-s|^{2H}$, как и утверждалось. Из доказанного следует, что
\begin{gather}\nonumber
|\mathcal{I}^{(1)}_{s,t}|\le |\mathcal{I}^{(0)}_s||B^{(i_1)}_{s,t}| + 
M_1|t-s|^{2H} \le \\ \nonumber
\le \bigl|\varphi\bigl(\widehat X^{(c)}_{cs}\bigr)\bigr|\|B^{(i_1)}\|_H|t-s|^H + 
M_1|t-s|^{H} \le \\ \label{bounds_3_60}
\le (\|\varphi\|_{\infty}\|B\|_H + M_1)|t-s|^H =: \widetilde{M}_1|t-s|^H.
\end{gather}
Легко видеть, что проведенные выше выкладки (\ref{bounds_3_56}) --- 
(\ref{bounds_3_60}) не зависят от значения $i_1$, т.е. справедливы и дают одни 
и те же оценки для любого $i_1=\overline{0,d}$.

Рассмотрим теперь $k=2$. Используя доказанное выше, 
предложение~\ref{hairer_th_4_10} и рекуррентное соотношение 
$\mathcal{I}^{(k)}_t = \int_0^t \mathcal{I}^{(k-1)}_r dB^{(i_k)}_r$, получим:
\begin{gather}\nonumber
\left|\mathcal{I}^{(2)}_{s,t} - \mathcal{I}^{(1)}_s B^{(i_2)}_{s,t}\right|\;\le
\; \left|(\mathcal{I}^{(1)})'_s\mathbb{B}^{(i_2)}_{s,t}\right| + \\ \nonumber
+ C\Bigl(\|B^{(i_2)}\|_H \cdot
\|(\mathcal{I}^{(1)})_{s,t} - (\mathcal{I}^{(1)})'_s B^{(i_2)}_{s,t}\|_{2H} + 
\|\mathbb{B}^{(i_2)}\|_{2H}\cdot\|(\mathcal{I}^{(1)})'\|_H\Bigr)|t-s|^{3H}= \\ 
\nonumber
= |\mathcal{I}^{(0)}_s\mathbb{B}^{(i_2)}_{s,t}| + 
C\Bigl(\|B^{(i_2)}\|_H \|\mathcal{I}^{(1)}_{s,t} - \mathcal{I}^{(0)}_s 
B^{(i_2)}_{s,t}\|_{2H} + 
\|\mathbb{B}^{(i_2)}\|_{2H}\|\mathcal{I}^{(0)}\|_H\Bigr)|t-s|^{3H} \le \\ 
\nonumber
\le\|\varphi\|_\infty\|\mathbb{B}\|_{2H}|t-s|^{2H} + C\Bigl(M_1\|B\|_H + 
\|\mathbb{B}\|_{2H} c^{H_{i_1}}\|D\varphi\|_\infty\|\widehat X^{(c)}\|_H\Bigr)
|t-s|^{2H} \le \\ \label{bounds_3_61}
\le M_2 |t-s|^{2H}.
\end{gather}
Здесь $M_2 = (\|\varphi\|_\infty + C\|D\varphi\|_\infty
\|\widehat X^{(c)}\|_H)\|\mathbb{B}\|_{2H} + C M_1 \|B\|_H$, а оценка на 
$\|\mathcal{I}^{(0)}\|_H$ была получена с помощью формулы конечных приращений:
\begin{equation}\label{bounds_3_62}
\mathcal{I}^{(0)}_{s,t} = \varphi\bigl(\widehat X^{(c)}_{ct}\bigr) - 
\varphi\bigl(\widehat X^{(c)}_{cs}\bigr) = 
D\varphi\Bigl(\widehat X^{(c)}_{cs} + \theta\bigl(\widehat X^{(c)}_{ct} - 
\widehat X^{(c)}_{cs}\bigl)\Bigr)\bigl(\widehat X^{(c)}_{ct} - 
\widehat X^{(c)}_{cs}\bigr).
\end{equation}
Так, мы получили 
$|\mathcal{I}^{(2)}_{s,t} - \mathcal{I}^{(1)}_s B^{(i_2)}_{s,t}|\le 
M_2|t-s|^{2H}$. Как и в случае $k = 1$, отсюда выводим, что 
$|\mathcal{I}^{(2)}_{s,t}|\le (\widetilde{M}_1\|B\|_H + M_2)|t-s|^{H} = 
\widetilde{M}_2|t-s|^{H}$, поскольку
\begin{gather}\nonumber
|\mathcal{I}^{(2)}_{s,t} - \mathcal{I}^{(1)}_sB^{(i_2)}_{s,t}|\ge 
|\mathcal{I}^{(2)}_{s,t}| - |\mathcal{I}^{(1)}_s|\cdot|B_{s,t}|\ge\\
\label{bounds_3_63}
\ge |\mathcal{I}^{(2)}_{s,t}| - \|\mathcal{I}^{(1)}\|_\infty \|B\|_H|t-s|^H\ge 
|\mathcal{I}^{(2)}_{s,t}| - \|\mathcal{I}^{(1)}\|_H \|B\|_H|t-s|^H.
\end{gather}
Легко видеть, что проведенные выше выкладки (\ref{bounds_3_61}) --- 
(\ref{bounds_3_63}) не зависят от значения $i_2$, т.е. справедливы и дают одни 
и те же оценки для любого $i_2=\overline{0,d}$.
\par

Переход доказывается аналогичным образом. Предположим, что утверждение 
выполнено для всех натуральных чисел, меньших $k$, и докажем его для $k+1$. 
По предположению индукции из предложения~\ref{hairer_th_4_10} будет следовать
\begin{gather}\nonumber
|\mathcal{I}^{(k+1)}_{s,t} - \mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t} - 
\mathcal{I}^{(k-1)}_s\mathbb{B}^{(i_{k+1})}_{s,t}| \le\\ \nonumber
\le C\Bigl(\|B^{(i_{k+1})}\|_H \cdot\|\mathcal{I}^{(k)}_{s,t} - 
\mathcal{I}^{(k-1)}_s B^{(i_{k+1})}_{s,t}\|_{2H} + 
\|\mathbb{B}^{(i_{k+1})}\|_{2H}\cdot\|\mathcal{I}^{(k-1)}\|_H\Bigr)
|t-s|^{3H} \le \\
\label{bounds_3_64}
\le C (M_k \|B\|_H+\widetilde{M}_{k-1}\|\mathbb{B}\|_{2H})|t-s|^{2H}.
\end{gather}
В то же время,
\begin{gather}\nonumber
|\mathcal{I}^{(k+1)}_{s,t} - \mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t} - 
\mathcal{I}^{(k-1)}_s\mathbb{B}^{(i_{k+1})}_{s,t}| \ge 
|\mathcal{I}^{(k+1)}_{s,t} - \mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t}| - 
|\mathcal{I}^{(k-1)}_s|\cdot|\mathbb{B}^{(i_{k+1})}_{s,t}|\ge\\ \nonumber
\ge|\mathcal{I}^{(k+1)}_{s,t} - \mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t}| - 
\|\mathcal{I}^{(k-1)}\|_{\infty}\|\mathbb{B}^{(i_{k+1})}\|_{2H}|t-s|^{2H}\ge\\
\ge |\mathcal{I}^{(k+1)}_{s,t} - \mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t}| - 
\widetilde{M}_{k-1}\|\mathbb{B}\|_{2H}|t-s|^{2H}.
\end{gather}
Таким образом,
\begin{equation}
|\mathcal{I}^{(k+1)}_{s,t} - \mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t}| \le 
M_{k+1}|t-s|^{2H},
\end{equation}
где $M_{k+1} = (C+1)\|\mathbb{B}\|_{2H}\widetilde{M}_{k-1} + C\|B\|_H M_k$. 
Осталось заметить, что
\begin{gather}\nonumber
M_{k+1}|t-s|^H\ge M_{k+1}|t-s|^{2H}\ge |\mathcal{I}^{(k+1)}_{s,t} - 
\mathcal{I}^{(k)}_s B^{(i_{k+1})}_{s,t}|\ge\\ \nonumber
\ge |\mathcal{I}^{(k+1)}_{s,t}| - |\mathcal{I}^{(k)}_s|\cdot
|B^{(i_{k+1})}_{s,t}| \ge 
|\mathcal{I}^{(k+1)}_{s,t}| - \|\mathcal{I}^{(k)}\|_\infty
\|B^{(i_{k+1})}\|_H|t-s|^H \ge\\ 
\ge |\mathcal{I}^{(k+1)}_{s,t}| - \widetilde{M}_k\|B\|_H|t-s|^H.
\end{gather}
Отсюда
\begin{equation}\label{bounds_3_68}
|\mathcal{I}^{(k+1)}_{s,t}| \le (M_{k+1}+\widetilde{M}_{k}\|B\|_H)|t-s|^H =: 
\widetilde{M}_{k+1}|t-s|^H,
\end{equation}
что и требовалось. Причем, как легко видеть, что проведенные выше 
выкладки (\ref{bounds_3_64}) --- (\ref{bounds_3_68}) не зависят от значения 
$i_{k+1}$, т.е. справедливы и дают одни и те же оценки для любого 
$i_{k+1}=\overline{0,d}$. Лемма доказана.
\bigskip

Из леммы следует, что
$$(\mathcal{I}^{(k)})' = \mathcal{I}^{(k-1)},  \ 
R^{\mathcal{I}^{(k)}}_{s,t} = (\mathcal{I}^{(k)})_{s,t} - 
(\mathcal{I}^{(k)})'_s B^{(i)}_{s,t} = 
\mathcal{I}^{(k)}_{s,t} - \mathcal{I}^{(k-1)}_s B^{(i)}_{s,t},$$
$$\|R^{\mathcal{I}^{(k)}}_{s,t}\|_{2H}\le M_k, \ 
\|\mathcal{I}^{(k)}\|_H \le \widetilde{M}_k,$$
а также $\|\mathcal{I}^{(k)}\|_{\infty}\le \widetilde{M}_k$, поскольку 
$\max_{t\in[0,1]}|\mathcal{I}^{(k)}_t| = 
\max_{t\in[0,1]}|\mathcal{I}^{(k)}_{0,t}|$.

Полученные рекуррентные соотношения 
\begin{equation}
M_{k+1} = (C+1)\|\mathbb{B}\|_{2H}\widetilde{M}_{k-1} + 
C\|B\|_H M_k,\quad \widetilde{M}_{k+1} = M_{k+1}+\widetilde{M}_{k}\|B\|_H
\end{equation}
с начальными условиями
\begin{gather}\nonumber
M_1 = \Bigl(C(\|D^2\varphi\|_{\infty}\|f\|_\infty + 
\|D\varphi\|_{\infty}\|Df\|_{\infty})\|\widehat X^{(c)}\|_H + 
\|D\varphi\|_{\infty}\|f\|_{\infty}\Bigr)\|\mathbb{B}\|_{2H} +\\
 +C\left(\frac{1}{2}\|D^2\varphi\|_{\infty}\|\widehat X^{(c)}\|^2_H + 
 \|D\varphi\|_{\infty}\|R^{\widehat X^{(c)}}\|_{2H}\right)\|B\|_{H},\\
\widetilde{M}_1 = \|\varphi\|_{\infty}\|B\|_H + M_1,\\
M_2 = (\|\varphi\|_\infty + C\|D\varphi\|_\infty\|\widehat X^{(c)}\|_H)
\|\mathbb{B}\|_{2H} + C M_1 \|B\|_H,\\
\widetilde{M}_2 = \widetilde{M}_1\|B\|_H + M_2
\end{gather}
позволяют последовательно вычислить константы $M_k, \widetilde{M}_k$. 
Очевидная индукция по $k$ показывает, что $M_k = M_k(\|B\|_H, 
\|\mathbb{B}\|_{2H}, \|\widehat X^{(c)}\|_{H}, 
\|R^{\widehat X^{(c)}}\|_{2H})$, $\widetilde{M}_k = 
\widetilde{M}_k(\|B\|_H, \|\mathbb{B}\|_{2H}, \|\widehat X^{(c)}\|_{H}, 
\|R^{\widehat X^{(c)}}\|_{2H})$ являются многочленами с постоянными 
положительными коэффициентами, причем $\|B\|_H$ и $\|\mathbb{B}\|_{2H}$ 
входят в одночлены в степени не выше $k$, $\|\widehat X^{(c)}\|_H$ --- в 
степени не выше $2$, а $\|R^{\widehat X^{(c)}}\|_{2H}$~--- в степени не 
выше $1$. Пусть $\widetilde{\gamma}_k$ --- максимальный коэффициент многочлена 
$ \widetilde{M}_k(\|B\|_H, \|\mathbb{B}\|_{2H}, \|\widehat X^{(c)}\|_{H}, 
\|R^{\widehat X^{(c)}}\|_{2H})$ и пусть 
\begin{equation}
\E\Bigl(\|B\|^{i_1}_H \|\mathbb{B}\|^{i_2}_{2H}\|\widehat X^{(c)}\|^{j_1}_{H} 
\|R^{\widehat X^{(c)}}\|^{j_2}_{2H}\Bigr) = 
\max\limits_{\substack{i, i' = \overline{0,k},\\ j = 0,1,2,\\ j' = 0,1}}
\E\Bigl(
\|B\|^{i}_H \|\mathbb{B}\|^{i'}_{2H}\|\widehat X^{(c)}\|^{j}_{H} 
\|R^{\widehat X^{(c)}}\|^{j'}_{2H}\Bigr).
\end{equation}
Тогда взяв супремум и математическое ожидание от обеих частей неравенства 
$|\mathcal{I}^{(k)}_{s,t}|\le \widetilde{M}_k|t-s|^H$, получим:
\begin{equation}
\sup_{0\le s< t\le 1}\E|\mathcal{I}^{(k)}_{s,t}|\le \E\widetilde{M}_k \le 
2k^2\cdot\widetilde{\gamma}_k\cdot\E\Bigl(
\|B\|^{i_1}_H \|\mathbb{B}\|^{i_2}_{2H}\|\widehat X^{(c)}\|^{j_1}_{H} 
\|R^{\widehat X^{(c)}}\|^{j_2}_{2H}\Bigr).
\end{equation}
Согласно неравенству Коши-Буняковского
\begin{gather}\nonumber
\E\Bigl(\|B\|^{i_1}_H \|\mathbb{B}\|^{i_2}_{2H}\|\widehat X^{(c)}\|^{j_1}_{H} 
\|R^{\widehat X^{(c)}}\|^{j_2}_{2H}\Bigr) \le \\ \nonumber
\le \Bigl(\E\bigl(\|B\|^{2i_1}_H \|\mathbb{B}\|^{2i_2}_{2H}\bigr)\cdot 
\E\bigl(\|\widehat X^{(c)}\|^{2j_1}_{H}
\|R^{\widehat X^{(c)}}\|^{2j_2}_{2H}\bigr)\Bigr)^{1/2}\le \\
\le  \Bigl(\E\|B\|^{4i_1}_H\cdot \E\|\mathbb{B}\|^{4i_2}_{2H}\cdot 
\E\|\widehat X^{(c)}\|^{4j_1}_{H}\cdot 
\E\|R^{\widehat X^{(c)}}\|^{4j_2}_{2H}\Bigr)^{1/4}.
\end{gather}
Таким образом,
\begin{gather} \nonumber
\E\sup_{0\le s< t\le 1}\left|
\int_s^t\int_0^{t_{k-1}}\ldots\int_0^{t_1}
\varphi\bigl(\widehat X^{(c)}_{c r}\bigr)\,
dB^{(i_1)}_{r}\ldots dB^{(i_{k-2})}_{t_{k-2}}dB^{(i_{k-1})}_{t_{k-1}}\right|\le\\ 
\label{bounds_3_85}
\le 2k^2\cdot\widetilde{\gamma}_k \cdot 
\Bigl(\E\|B\|^{4i_1}_H\cdot \E\|\mathbb{B}\|^{4i_2}_{2H}\cdot 
\E\|\widehat X^{(c)}\|^{4j_1}_{H}\cdot 
\E\|R^{\widehat X^{(c)}}\|^{4j_2}_{2H}\Bigr)^{1/4}.
\end{gather}
Согласно \cite[лемма 7.4]{Nualart} любой момент порядка $p\ge1$ случайной 
величины~$\|B\|_H$ конечен; в частности, $\E\|B\|^{4i_1}_H<\infty$. То же верно 
и для случайной величины $\|\mathbb{B}\|_{2H}$ ввиду предложения 
\ref{double_B_is_geometric_rough_path}, т.е., в частности, 
$\E\|\mathbb{B}\|^{4i_2}_{2H}<\infty$. Из предложения~\ref{hairer_prop_8_3} 
следует, что существуют универсальные константы $c_1$, $c_2$ такие, что
\begin{gather}\label{y_bound}
\|\widehat X^{(c)}\|_H \le c_1 \left(\|\widehat B^{(c)}\|_H + 
\|\widehat{\mathbb{B}}^{(c)}\|^{1/2}_{2H} + \|\widehat B^{(c)}\|^{1/H}_H +
\|\widehat{\mathbb{B}}^{(c)}\|^{1/(2H)}_{2H}\right),\\
\label{ry_bound}
\|R^{\widehat X^{(c)}}\|_{2H} \le c_{2}\left(\|\widehat B^{(c)}\|_H^2 + 
\|\widehat{\mathbb{B}}^{(c)}\|_{2H} + \|\widehat B^{(c)}\|^{1+\frac{1}{H}}_H + 
\|\widehat{\mathbb{B}}^{(c)}\|^{\frac{1}{2}+\frac{1}{2H}}_{2H} \right).
\end{gather}
Из неравенства Гельдера следует, что конечны любые моменты $\|B\|_H$, 
$\|\mathbb{B}\|_{2H}$ порядка $q\in (0,1)$: $\E\|B^{(c)}\|^q_H \le 
\Bigl(\E\|B^{(c)}\|_H\Bigr)^q \Bigl(\E 1^{1/(1-q)}\Bigr)^{1-q}<\infty$ 
(и аналогично $\E\|\mathbb{B}^{(c)}\|^{q}_{2H} < \infty$). Поэтому из оценок 
(\ref{y_bound}) и (\ref{ry_bound}) и неравенств о средних следует конечность 
моментов
\begin{equation}
\E\|X^{(c)}\|^{4j_1}_H < \infty, \qquad \E\|R^{X^{(c)}}\|^{4j_2}_{2H} < \infty.
\end{equation}
Это, в свою очередь, завершает доказательство того, что правая часть 
(\ref{bounds_3_85}) конечна. Теорема \ref{chapter_3_asympt_development} 
доказана.

\remark
Рассмотрим случай, когда все показатели Харста $H_i$, $i=\linebreak=1,\ldots,d,$ 
равны $1/2$. В этом случае уравнение (\ref{chapter3_main_eq_compact}) можно 
рассматривать как уравнение Стратоновича. Учитывая связь уравнений Ито и 
Стратоновича \cite[предложение 2.4]{Levakov_book}, данное уравнение можно 
свести к уравнению Ито
\begin{equation} \label{eq_Ito}
dX_t=\tilde f_0(X_t)dt+\hat f(X_t)dW_t,
\end{equation}
где $\hat f$ -- матрица, составленная из вектор-столбцов $f_1,\ldots, f_d$, 
$W_t$ -- $d$-мерное броуновское движение,
$$
\tilde f_0(X)=f_0(X)-\textrm{col}(\rho_1(X),\ldots, \rho_n(X)),
$$
$$
\rho_j(X)=\frac{1}{2} \sum_{i=1}^n \sum_{k=1}^n 
\frac{\partial \hat f_{ji}(X)}{\partial x_k} \hat f_{ki}(X),\quad j=1,\ldots, d,
$$
где $\text{col}(a_1, \ldots, a_n)$ --- вектор-столбец из компонент $a_1,\ldots, a_n$.
Так как решение уравнения (\ref{eq_Ito}) обладает марковским свойством 
\cite[теорема~7.1.2]{OksendalBook}, то, полагая $N=2$ в соотношении 
(\ref{bounds_3_43}), получим, что функция $u(x,t)=\textup{\textbf{P}}_t g(x)$
удовлетворяет обратному уравнению Колмогорова
$$
\frac{\partial u}{\partial t}=\mathcal{A}u
$$
с дифференциальным оператором
$$
\mathcal{A}=\sum_{j=1}^n \tilde f_{0j}(\cdot) \frac{\partial}{\partial x_j} + 
\frac{1}{2} \sum_{k=1}^d \biggl( 
\sum_{j=1}^n f_{kj}(\cdot) \frac{\partial}{\partial x_j} \biggr)^2.
$$
Отметим, что даже для простейшего одномерного уравнения
$
dX_t=dB^\alpha_t
$
при $\alpha \in (1/3, 1/2) \cup (1/2, 1)$ решение $X_t=B^\alpha_t$ не является 
семимартингалом, и следовательно, не обладает марковским свойством, которое 
является ключевым при выводе уравнений Колмогорова 
\cite[теорема 8.1.1]{OksendalBook}.


\section{Математические ожидания повторных интегралов от дробных броуновских 
движений}

В данном разделе мы вычислим математические ожидания повторных интегралов 
возникающих их разложений для $\textup{\textbf{P}}_t g(x)$ в теореме 
\ref{chapter_3_asympt_development}, предполагая, что  $H_i\ge H^*>1/2$ для всех 
$i=0,\ldots, d.$


\begin{theorem}\label{theorem_5_1}
Пусть $I_{m}=(i_1,\ldots, i_{m})\in\{1,\ldots, d\}^{m}$, $m\in\mathbb{N}$.
\begin{enumerate}
\item Если $m=2k-1$, $k\in\mathbb{N},$ то
$$\E\biggl(\int_{\Delta^{2k-1}[0,1]} dB^{(I_{2k-1})}\biggr) = 0.$$
\item Если $m=2k$, $k\in\mathbb{N},$ то справедливо равенство
\begin{gather*}
\E\biggl(\int_{\Delta^{2k}[0,1]} dB^{(I_{2k})}\biggr)= \\
= \frac{1}{k! 2^{2k}}\sum_{\sigma\in{S}_{2k}}\biggl(
\prod_{l=1}^k(H_{i_{\sigma(2l)}} {+} H_{i_{\sigma(2l-1)}} {-} 1)
(H_{i_{\sigma(2l)}} {+} H_{i_{\sigma(2l-1)}})\times \\ 
\times \!\!\!\!\!\!\!\int\limits_{\Delta^{2k}[0,1]}\!
\prod_{l=1}^k\delta_{\substack{i_{\sigma(2l)}, i_{\sigma(2l-1)}}}
|t_{\sigma(2l)}{-}t_{\sigma(2l-1)}|^{H_{i_{\sigma(2l)}} 
{+} H_{i_{\sigma(2l-1)}} {-} 2}dt_1\ldots dt_{2k} \biggr),
\end{gather*}
где $\delta$ --- символ Кронекера, $S_{2k}$ --- группа подстановок.
\end{enumerate}
\end{theorem}

\proof Первую часть легко доказать, используя свойство симметрии распределения 
дробного броуновского движения: если $B^{(i)}$ --- дробное броуновское движение 
с индексом Харста $H_i,$ то $-B^{(i)}$ --- также дробное броуновское движение 
с тем же индексом Харста. Таким образом, получим
\begin{gather*}
\E\biggl(\int_{\Delta^{2k-1}[0,1]} dB^{(I_{2k-1})}\biggr) = 
\E\biggl(\int_{\Delta^{2k-1}[0,1]} d\left(-B^{(I_{2k-1})}\right)\biggr)= \\
= (-1)^{2k-1}\E\biggl(\int_{\Delta^{2k-1}[0,1]} dB^{(I_{2k-1})}\biggr) =
 -\E\biggl(\int_{\Delta^{2k-1}[0,1]} dB^{(I_{2k-1})}\biggr),
\end{gather*}
что доказывает первое утверждение теоремы.

Докажем вторую часть теоремы.
Обозначим через $B^{(N)}$, $N\in\mathbb{N},$ последовательность приближений 
к $B$ на диадных разбиениях~$\mathcal{P}^{(N)}_{dyad} =\linebreak= 
\left\{t^{(N)}_k = \frac{k}{2^N},\; k=\overline{0, 2^N}\right\}$ отрезка 
$[0,1]$, определяемых следующей формулой:
$$B^{(N)}_t  = B_{t^{(N)}_{k-1}} + 
2^N\Bigl(t - t^{(N)}_{k-1}\Bigr)\left(B_{t^{(N)}_k} - B_{t^{(N)}_{k-1}}\right), \ 
t\in\Bigl[t^{(N)}_{k-1}, t^{(N)}_{k}\Bigr),\; k=1,\ldots, N.$$
По теореме о мажорируемой сходимости (предложение~\ref{dominated_convergence_th}) 
ввиду \cite[следствие 20]{Coutin_Qian}, получим:
$$\E\biggl(\int_{\Delta^{2k}[0,1]} dB^{(I_{2k})}\biggr) = 
\lim_{N\to\infty}\E\biggl(\int_{\Delta^{2k}[0,1]} (dB^{(N)})^{(I_{2k})}\biggr).$$
Поскольку функции $B^{(N)}$ абсолютно непрерывны п.н., то справедливы 
соотношения
\begin{gather} \nonumber
\int_{\Delta^{2k}[0,1]} (dB^{(N)})^{(I_{2k})}=\\
\nonumber
= \int_0^1\int_0^{t_{2k}}\ldots \int_0^{t_2} d\bigl(B^{(N)}_{t_1}\bigr)^{(i_1)} 
\ldots d\bigl(B^{(N)}_{t_{2k-1}}\bigr)^{(i_{2k-1})}
d\bigl(B^{(N)}_{t_{2k}}\bigr)^{(i_{2k})}= \\ \nonumber
= \int_0^1\int_0^{t_{2k}}\ldots \int_0^{t_2} 
\frac{d\bigl(B^{(N)}_{t_1}\bigr)^{(i_1)}}{dt_1}\,dt_1 \ldots 
\frac{d\bigl(B^{(N)}_{t_{2k-1}}\bigr)^{(i_{2k-1})}}{dt_{2k-1}}\,dt_{2k-1} 
\frac{d\bigl(B^{(N)}_{t_{2k}}\bigr)^{(i_{2k})}}{dt_{2k}}\, dt_{2k}=\nonumber\\
= \int_{\Delta^{2k}[0,1]} \frac{d\bigl(B^{(N)}_{t_1}\bigr)^{(i_1)}}{dt_1}\ldots 
\frac{d\bigl(B^{(N)}_{t_{2k}}\bigr)^{(i_{2k})}}{dt_{2k}}\, dt_1\ldots dt_{2k}. 
\label{eq_multiple}
\end{gather}

Нам потребуется следующее утверждение, которое может быть доказано 
непосредственно путем разложения в ряд производящей функции моментов 
гауссовского случайного вектора.

\begin{lemma}[\cite{Baudoin_Coutin}] \label{gaussian_lemma}
Для центрированного гауссовского вектора $G=\linebreak=(G_1,\ldots, G_{2k})$ 
справедливо равенство
$$\E (G_1\ldots G_{2k}) = \frac{1}{k!2^k}\sum_{\sigma\in{S}_{2k}}
\prod_{l=1}^k \E(G_{\sigma(2l)}G_{\sigma(2l-1)}).$$
\end{lemma}


Применяя лемму \ref{gaussian_lemma} к равенству (\ref{eq_multiple}), получим:
\begin{gather}\nonumber
\E\biggl(\int_{\Delta^{2k}[0,1]} (dB^{(N)})^{(I_{2k})}\biggr)= \\ \nonumber
= \int_{\Delta^{2k}[0,1]} 
\E\biggl( \frac{d\bigl(B^{(N)}_{t_1}\bigr)^{(i_1)}}{dt_1}\ldots 
\frac{d\bigl(B^{(N)}_{t_{2k}}\bigr)^{(i_{2k})}}{dt_{2k}} \biggr) 
dt_1\ldots dt_{2k}=\\
=\frac{1}{k! 2^k}\sum_{\sigma\in{S}_{2k}} 
\int_{\Delta^{2k}[0,1]} \prod_{l=1}^k 
\E\biggl( 
\frac{d\bigl(B^{(N)}_{t_{\sigma(2l)}}\bigr)^{(i_{\sigma(2l)})}}{dt_{\sigma(2l)}}
\cdot
\frac{d\bigl(B^{(N)}_{t_{\sigma(2l-1)}}\bigr)^{(i_{\sigma(2l-1)})}}{dt_{\sigma(2l-1)}}
\biggr) dt_1\ldots dt_{2k}.\label{th_5_1_applying_lemma}
\end{gather}
Рассмотрим отдельно каждый множитель в соотношении (\ref{th_5_1_applying_lemma}). 
Пусть $t_{\sigma(2l)}\in\linebreak\in[t^{(N)}_u, t^{(N)}_{u+1})$ и 
$t_{\sigma(2l-1)}\in[t^{(N)}_v, t^{(N)}_{v+1})$ для некоторых 
$u,v\in\{0, 1, \ldots, 2^N-1\}$ (для упрощения обозначений будем иногда опускать 
верхний индекс $(N)$). Используя представление $B^{(N)},$ можем записать:
\begin{gather*}
\E\biggl( 
\frac{d\bigl(B^{(N)}_{t_{\sigma(2l)}}\bigr)^{(i_{\sigma(2l)})}}{dt_{\sigma(2l)}}
\cdot
\frac{d\bigl(B^{(N)}_{t_{\sigma(2l-1)}}\bigr)^{(i_{\sigma(2l-1)})}}{dt_{\sigma(2l-1)}} 
\biggr)= \\
= 2^{2N}\E \Bigl(B^{(i_{\sigma(2l)})}_{t_{u+1}} - B^{(i_{\sigma(2l)})}_{t_u}\Bigr) 
\Bigl(B^{(i_{\sigma(2l-1)})}_{t_{v+1}} - B^{(i_{\sigma(2l-1)})}_{t_v}\Bigr).
\end{gather*}
Используя неравенство Коши-Буняковского, однородность приращений дробного 
Броуновского движения, получим неравенство
\begin{gather}\nonumber
\left|\E \Bigl(B^{(i_{\sigma(2l)})}_{t_{u+1}} - B^{(i_{\sigma(2l)})}_{t_u}\Bigr) 
\Bigl(B^{(i_{\sigma(2l-1)})}_{t_{v+1}} - B^{(i_{\sigma(2l-1)})}_{t_v}\Bigr)\right|
\le \nonumber  \\ 
\le \sqrt{|t_{u+1} - t_u|^{2H_{i_{\sigma(2l)}}} |t_{v+1} - t_v|^{2H_{i_{\sigma(2l-1)}}}}
\le 2^{-2NH^*}. \label{theorem_5_1_cauchy_shwarz}
\end{gather}


Если $|u-v| > 1,$ то легко вывести следующее равенство:
\begin{gather}\nonumber
\E \Bigl(B^{(i_{\sigma(2l)})}_{t_{u+1}} - B^{(i_{\sigma(2l)})}_{t_u}\Bigr) 
\Bigl(B^{(i_{\sigma(2l-1)})}_{t_{v+1}} - B^{(i_{\sigma(2l-1)})}_{t_v}\Bigr)= 
\nonumber \\ 
= \frac{1}{2} \delta_{i_{\sigma(2l)}, i_{\sigma(2l-1)}}(H_{i_{\sigma(2l)}} + 
H_{i_{\sigma(2l-1)}} - 1)(H_{i_{\sigma(2l)}} + H_{i_{\sigma(2l-1)}})\times 
\nonumber\\ 
\times \iint_{[t^{(N)}_u, t^{(N)}_{u+1})\times [t^{(N)}_v, t^{(N)}_{v+1})}
|x-y|^{H_{i_{\sigma(2l)}} + H_{i_{\sigma(2l-1)}}-2}dxdy. 
\label{th_5_1_exact_approach}
\end{gather}
Введем следующее обозначение:
$$R^{(N)}_{u,v} = [t^{(N)}_u, t^{(N)}_{u+1})\times [t^{(N)}_v, t^{(N)}_{v+1}),$$
$$f^{(l)}_{u,v}(t) = \textbf{1}_{R^{(N)}_{u,v}}(t_{\sigma(2l)}, t_{\sigma(2l-1)}) 
\E\Bigl(B^{(i_{\sigma(2l)})}_{t_{u+1}} - B^{(i_{\sigma(2l)})}_{t_u}\Bigr)
\Bigl(B^{(i_{\sigma(2l-1)})}_{t_{v+1}} - B^{(i_{\sigma(2l-1)})}_{t_v}\Bigr).$$  

Используя соотношение (\ref{th_5_1_applying_lemma}), получим:
\begin{gather*}
\E\biggl(\int_{\Delta^{2k}[0,1]} (dB^{(N)})^{(I_{2k})}\biggr)= \\
=\frac{2^{2Nk}}{k! 2^k}\sum_{\sigma\in{S}_{2k}} \int_{\Delta^{2k}[0,1]} 
\prod_{l=1}^k \sum_{u,v=0}^{2^N-1}
\textbf{1}_{R^{(N)}_{u,v}}(t_{\sigma(2l)}, t_{\sigma(2l-1)})\times \\ 
\nonumber 
\times \E\Bigl(B^{(i_{\sigma(2l)})}_{t_{u+1}} - B^{(i_{\sigma(2l)})}_{t_u}\Bigr)
\Bigl(B^{(i_{\sigma(2l-1)})}_{t_{v+1}} - B^{(i_{\sigma(2l-1)})}_{t_v}\Bigr) 
dt_1\ldots dt_{2k}=\\
=\frac{2^{2Nk}}{k! 2^k}\sum_{\sigma\in{S}_{2k}} \int_{\Delta^{2k}[0,1]} 
\prod_{l=1}^k \left( \sum_{\substack{u,v=0\\ |u-v|> 1}}^{2^N-1} f^{(l)}_{u,v}(t) + 
\sum_{\substack{u,v=0\\ |u-v|\le 1}}^{2^N-1}f^{(l)}_{u,v}(t)\right) 
dt_1\ldots dt_{2k}= \\
= \frac{2^{2Nk}}{k! 2^k}\sum_{\sigma\in{S}_{2k}} \int_{\Delta^{2k}[0,1]} 
\prod_{l=1}^k  \sum_{\substack{u,v=0\\ |u-v|> 1}}^{2^N-1} f^{(l)}_{u,v}(t)\; 
dt_1\ldots dt_{2k}+ \\
+  \frac{2^{2Nk}}{k! 2^k}\sum_{\sigma\in{S}_{2k}} \int_{\Delta^{2k}[0,1]} 
\sum_{\alpha=0}^{k-1}\sum_{\pi\in{S}_k}\biggl(\prod_{l=1}^\alpha
\sum_{\substack{u,v=0\\ |u-v|> 1}}^{2^N-1} f^{(\pi(l))}_{u,v}(t)\biggr)\times \\
\times \biggl( \prod_{l=\alpha+1}^k \sum_{\substack{u,v=0\\ |u-v|\le 1}}^{2^N-1}
f^{(\pi(l))}_{u,v}(t)\biggr) dt_1\ldots dt_{2k}:= I^{(N)}_1 + I^{(N)}_2.
\end{gather*}

Сейчас утверждение теоремы является прямым следствием следующей леммы.

\begin{lemma}\label{theorem_5_1_technic_lemma}
В предыдущих обозначениях пусть
\begin{gather*}
I^{(N)}_1 = \frac{2^{2Nk}}{k! 2^k}\sum_{\sigma\in{S}_{2k}} 
\int_{\Delta^{2k}[0,1]} \prod_{l=1}^k  
\sum_{\substack{u,v=0\\ |u-v|> 1}}^{2^N-1} f^{(l)}_{u,v}(t)\; dt_1\ldots dt_{2k},\\
I^{(N)}_2 = \frac{2^{2Nk}}{k! 2^k}\sum_{\sigma\in{S}_{2k}} 
\int_{\Delta^{2k}[0,1]} \sum_{\alpha=0}^{k-1}
\sum_{\pi\in{S}_k}\biggl(\prod_{l=1}^\alpha\sum_{\substack{u,v=0\\ |u-v|> 1}}^{2^N-1} 
f^{(\pi(l))}_{u,v}(t)\biggr)\times
\end{gather*}
\begin{gather*}
\times \biggl( \prod_{l=\alpha+1}^k \sum_{\substack{u,v=0\\ |u-v|\le 1}}^{2^N-1}
f^{(\pi(l))}_{u,v}(t)\biggr) dt_1\ldots dt_{2k}.
\end{gather*}
Тогда $\lim\limits_{N\to\infty} I^{(N)}_2 = 0$ и
\begin{gather*}
\lim_{N\to\infty} I^{(N)}_1 = \frac{1}{k! 2^{2k}}\sum_{\sigma\in{S}_{2k}}
\left(\prod_{l=1}^k(H_{i_{\sigma(2l)}} {+} H_{i_{\sigma(2l-1)}} {-} 1)
(H_{i_{\sigma(2l)}} {+} H_{i_{\sigma(2l-1)}})\right)\times \\
\times \int_{\Delta^{2k}[0,1]}
\prod_{l=1}^k\delta_{\substack{i_{\sigma(2l)}, i_{\sigma(2l-1)}}}
|t_{\sigma(2l)}{-}t_{\sigma(2l-1)}|^{H_{i_{\sigma(2l)}} {+} H_{i_{\sigma(2l-1)}} {-} 2}
dt_1\ldots dt_{2k}.
\end{gather*}
\end{lemma}

Доказательство данной леммы опирается на теорему о среднем и неравенство Гелдьера,
оно приведено в статье [\ref{StochasticAnalysis}]. Теорема доказана.


\remark\label{theorem_5_1_remark} Рассмотрим математические ожидания повторных 
интегралов следующего вида
$$
J(I^{(m)}, t):=\E\biggl(\int_{\Delta^{m}[0,t]} dB^{(I_{m})}\biggr)
$$
для произвольных индексов $I_{m}=(i_1,\ldots, i_{m})\in\mathbb{N}_d^m.$  Без 
ограничения общности будем считать, что $i_1,\ldots, i_k>0$, а $i_j=0$ для всех 
$j>k.$ Тогда следующее равенство может быть получено изменением порядка 
интегрирования:
$$
J(I^{(m)}, t)= \int_0^t J(\widetilde{I^{(k)}}, \tau)
\frac{(t-\tau)^{m-k-1}}{(m-k-1)!}d\tau,
$$
где $\widetilde{I^{(k)}}=(i_1,\ldots, i_k)$ и $J(\widetilde{I^{(k)}}, \tau)$ 
могут быть вычислены, используя равенство~(\ref{bounds_3_48}) и 
теорему \ref{theorem_5_1}.

\example Рассмотрим следующее одномерное уравнение:
$$dX_t =  b(X_t) \,dt + \sigma(X_t)\,dB^H_t,$$
в котором $B^H_t$ --- одномерное дробное броуновское движение с индексом 
Харста $H\in(1/2,1)$, $b,\,\sigma\colon\mathbb{R}\to\mathbb{R}$ --- функции 
класса $C^4_b$. Обозначим $f(x) =\linebreak= (b(x), \sigma(x))$, 
$B_t = (t, B^H_t)$. Пусть также задано начальное условие $X_0 = x$. Выпишем 
несколько первых членов асимптотических разложений~(\ref{bounds_3_43}) 
(ограничимся $N=2$) для решений данного уравнения.

Используя теорему~\ref{theorem_5_1} и замечание~\ref{theorem_5_1_remark} можем 
вычислить повторные интегралы:
\begin{gather*}
\E\left(\int_{\Delta^1[0,1]}dB^{(0)}\right) = \int_0^1dt=1, \quad 
\E\left(\int_{\Delta^1[0,1]}dB^{(1)}\right) = \E\left(\int_0^1dB^H_t\right)=0,\\
\E\left(\int_{\Delta^2[0,1]}dB^{(0,0)}\right) = \int_0^1 dt_2 \int_0^{t_2} dt_1=
\int_0^1t_2dt_2 = \frac{1}{2},\\
\E\left(\int_{\Delta^2[0,1]}dB^{(0,1)}\right) = 
\E\left(\int_{\Delta^2[0,1]}dB^{(1,0)}\right) = 
\int_0^1 \E\left(\int_0^{t_2}dB^H_{t_1}\right)dt_2 = 0,\\
\E\left(\int_{\Delta^2[0,1]}dB^{(1,1)}\right) = 
\E\left(\int_0^1 dB^H_{t_2}\int_0^{t_2} dB^H_{t_1}\right) = 
\frac{1}{1!2^2}\cdot 2\cdot 2H(2H-1)\times\\
\times\int_{\Delta^2[0,1]}|t_2 - t_1|^{2H-2}dt_1dt_2= 
H(2H-1)\int_0^1 dt_2\int_0^{t_2} (t_2-t_1)^{2H-2}dt_1=\\
= H\int_0^1 t_2^{2H-1}dt_2 = \frac{1}{2}.
\end{gather*}

Для данного уравнения имеем операторы $D^{(0)}_f = b(x)\, 
\frac{\partial}{\partial x}$ и $D^{(1)}_f = \sigma(x)\, 
\frac{\partial}{\partial x}$. Нетрудно вычислить, что
$$
D^{(0,0)}_f = b(x)Db(x)\,\frac{\partial}{\partial x} + 
b^2(x)\,\frac{\partial^2}{\partial x^2}, \quad D^{(1,1)}_f = 
\sigma(x)D\sigma(x)\,\frac{\partial}{\partial x} + 
\sigma^2(x)\,\frac{\partial^2}{\partial x^2}.
$$
Таким образом, согласно теореме~\ref{chapter_3_asympt_development}, для 
$\textbf{P}_t g(x)$, $g\in C^5_b (\mathbb{R}, \mathbb{R})$ справедливо 
асимптотическое разложение следующего вида:
\begin{gather*}
\textbf{P}_t g(x) = g(x) + t\,b(x)Dg(x) + 
\frac{1}{2} t^2 \left(b(x)Db(x)Dg(x) + b^2(x) D^2g(x)\right) +\\ 
+\frac{1}{2} t^{2H}\left(\sigma(x)D\sigma(x)Dg(x) + \sigma^2(x) D^2g(x)\right) +
 O\left(t^{3H}\right).
\end{gather*}


\section{Коммутативный случай}\label{commutative_case_section}

В данном разделе будем предполагать, что компоненты $f_i\colon \mathbb{R}^n\to 
\mathbb{R}^{n}$ правой части уравнения~(\ref{chapter3_main_eq_compact}) 
являются векторными полями из класса $C^{d+2}_b(\mathbb{R}^n, \mathbb{R}^n)$ 
такими, что справедливо равенство $D_f^{(i)} \circ D_f^{(j)} = D_f^{(j)} \circ D_f^{(i)}$, 
для любых $0\le i,j\le d$. 

Для каждого $i=\overline{0,d}$ обозначим через $(e^{tD_f^{(i)}})_{t\in\mathbb{R}}
\subset\mathcal{L}(\mathbb{R}^n, \mathbb{R}^n)$ семейство операторов, определяемое 
соотношением $e^{tD_f^{(i)}}(x) = X^x_t$ для любых $t\in\mathbb{R}$, $x\in\linebreak
\in\mathbb{R}^n$, в котором $X^x_t$ --- решение обыкновенного дифференциального 
уравнения
$$\frac{dX_t}{dt} = f_i(X_t)$$
с начальным условием $X_0 = x$.

\begin{proposition}\label{flow_proposition}
Для решения $X^x_t$ уравнения~\textup{(\ref{chapter3_main_eq_compact})}
c начальным условием $X_0 = x$ п. н. справедлива следующая формула:
$$X^x_t = F(x, B_t), \quad t \in [0, T], \; x\in\mathbb{R}^n,$$
в которой $F(x, y) = e^{y_0 D_f^{(0)}}\circ\ldots\circ e^{y_d D_f^{(d)}}$, 
$(x, y)\in\mathbb{R}^n\times\mathbb{R}^{d+1}$.
\end{proposition}

\proof Поскольку операторы $D_f^{(i)}$ коммутируют, то 
операторы~$(e^{t D_f^{(i)}})_{t\ge 0}$ также коммутируют. Для пары 
$(x,y)\in \mathbb{R}^n\times\mathbb{R}^{d+1}$ обозначим
$$
F(x,y) = \left(e^{y_0 D_f^{(0)}}\circ\ldots\circ e^{y_d D_f^{(d)}}\right)(x).
$$
Применяя формулу замены переменных, легко видеть, что процесс 
$\left(e^{B^{(d)}_t D_f^{(d)}}x\right)_{t\ge0}$ является решением уравнения
$dX_t = f_d(X_t)dB^{(d)}_t.$
Применяя формулу замены переменных еще раз и учитывая коммутативность 
операторов $D_f^{(d)}$, $D_f^{(d-1)}$, получим:
\begin{gather*}
d\left(e^{B^{(d-1)}_t D_f^{(d)-1}}(e^{B^{(d)}_t D_f^{(d)}}x)\right) = 
f_{d-1}\left(e^{B^{(d-1)}_t D_f^{(d-1)}}(e^{B^{(d)}_t D_f^{(d)}}x)\right)dB^{(d-1)}_t \\
+ f_{d}\left(e^{B^{(d-1)}_t D_f^{(d-1)}}(e^{B^{(d)}_t D_f^{(d)}}x)\right)dB^{(d)}_t.
\end{gather*}
И так далее. Путем последовательного применения формулы замены переменных, 
заключаем, что процесс $(F(x, B_t))_{t\ge0}$ удовлетворяет уравнению 
(\ref{chapter3_main_eq_compact}) с начальным условием $X_0=x.$
Таким образом, согласно теореме \ref{chapter_3_existance_th}, 
можем сделать вывод, что
$$X^{x}_t = F(x, B_t),\quad t\in [0,T],$$
п.н. Предложение доказано.

\begin{theorem}\label{chapter_3_commut_case_main_th}
Для любой функции $g\in C^{d+3}_b(\mathbb{R}^n, \mathbb{R})$ 
справедливо равенство
$$\E\left(g(X^{x}_t)\right) = \left(\exp\left(t D_f^{(0)} + 
\frac{1}{2}\sum_{i=1}^d t^{2H_i}(D_f^{(i)})^2\right) g\right)(x).$$
Другими словами, функция
$$\varphi(t,x) = \E \left(g(X^{x}_t)\right),$$
удовлетворяет дифференциальному уравнению в частных производных
\begin{equation}\label{chapter_3_commut_case_main_eq}
\frac{\partial \varphi}{\partial t} = D_f^{(0)}\varphi + 
\sum_{i=1}^d H_it^{2H_i - 1} (D_f^{(i)})^2 \varphi,
\end{equation}
с начальным условием
$$\varphi(0, x) = g(x).$$
\end{theorem}

\proof Применяя формулу Ито для дробного броуновского движения из \cite{Cheridito}, 
получим для $i > 0$:
$$\E\left(g(e^{B^{(i)}_t D_f^{(i)}}(x))\right) = g(x) + 
H_i \int_0^t s^{2H_i - 1} \E\left((D_f^{(i)})^2 
g(e^{B^{(i)}_s D_f^{(i)}} (x))\right) ds.$$
Если $i=0$, то для $B^{(0)}_t=t$ будем иметь следующее равенство
$$g(e^{B^{(0)}_t D_f^{(0)}}(x)) = g(x) + 
\int_0^t D_f^{(0)} g(e^{B^{(0)}_s D_f^{(0)}} (x)) ds.$$
Последние два равенства означают, что функции $u_i(t, x) = 
\E\left(g(e^{B^{(i)}_t D_f^{(i)}}(x))\right)$ являются решениями 
следующих уравнений: 
\begin{gather*}
\frac{\partial u_i}{\partial t} = H_i t^{2H_i - 1} (D_f^{(i)})^2 u_i,\quad i > 0,\\
\frac{\partial u_0}{\partial t} = D_f^{(0)} u_0,
\end{gather*}
следовательно,
\begin{gather*}
\E\left(g(e^{B^{(i)}_t D_f^{(i)}}(x))\right) = 
\left(\exp\left(\frac{1}{2}t^{2H_i} (D_f^{(i)})^2\right) g\right)(x),\quad i > 0,\\
g(e^{B^{(0)}_t D_f^{(0)}}(x)) = \left(\exp\left(t D_f^{(0)}\right) g\right)(x).
\end{gather*}
Согласно предложению \ref{flow_proposition}, п.н. верно равенство
$$X_t^{x} = (e^{B^{(0)}_t D_f^{(0)}}\circ\ldots\circ e^{B^{(d)}_t D_f^{(d)}}) (x).$$
Используя коммутативность операторов $D_f^{(i)}$ и $D_f^{(j)}$, можем записать, 
что
$$\E\left(g(X^{x}_t)\right) = \left(\exp\left(t D_f^{(0)} + 
\frac{1}{2}\sum_{i=1}^d t^{2H_i}(D_f^{(i)})^2\right) g\right)(x),$$
что, в свою очередь, доказывает теорему.

\remark Если компоненты правой части 
$f_i\in C^{d+2}_b(\mathbb{R}^n, \mathbb{R}^n)$ автономны, а операторы 
$D^{(i)}_f$ и $D^{(j)}_j$ коммутируют, то функция $\E(g(X^x_t))$ удовлетворяет
уравнению~(\ref{chapter_3_commut_case_main_eq}), которое является обобщением 
обратного уравнения Колмогорова \cite[теорема 8.1.1]{OksendalBook} на случай 
дробного броуновского движения с различными индексами Харста, вообще говоря 
отличными от $1/2$.

\example Рассмотрим следующее одномерное стохастическое дифференциальное 
уравнение:
$$dX_t = \sin X_t \,dt + 2H^{-1}\sin X_t \,dB^H_t,$$
в котором $B^H_t$ --- одномерное дробное броуновское движение с индексом 
Харста $H\in(1/3,1)$. Пусть также задано начальное условие 
$X_0 = x\in\mathbb{R}$.

Для данного уравнения имеем операторы $D^{(0)}_f = \sin x\, 
\frac{\partial}{\partial x}$ и $D^{(1)}_f =\linebreak= 2H^{-1}\sin x\, 
\frac{\partial}{\partial x}$. Нетрудно видеть, что указанное уравнение 
удовлетворяет коммутативному случаю, поскольку
$$D^{(0)}_f \circ D^{(1)}_f = 2H^{-1}\sin x \cos x\, \frac{\partial}{\partial x} + 
2H^{-1}\sin^2 x \,\frac{\partial^2}{\partial x^2} = D^{(1)}_f \circ D^{(0)}_f.$$
Непосредственные вычисления показывают, что $(D^{(1)}_f)^2 = 
H^{-1}\sin 2x \frac{\partial}{\partial x} +\linebreak+ 4H^{-2}\sin^2 x 
\frac{\partial^2}{\partial x^2}$. 
Поэтому, согласно теореме~\ref{chapter_3_commut_case_main_th}, функция 
$\varphi(t,x) = \E g(X^x_t)$ будет являться решением уравнения в частных 
производных
$$\frac{\partial\varphi}{\partial t} = \left(\sin x + t^{2H-1}\sin 2x\right)
\frac{\partial\varphi}{\partial x} + \frac{4t^{2H-1}}{H}\sin^2 x\, 
\frac{\partial^2\varphi}{\partial x^2}$$
с начальным условием $\varphi(0,x) = g(x)$.


\section*{Выводы}
\addcontentsline{toc}{section}{Выводы}

В данной главе диссертации исследовано асимптотическое поведение функционалов 
от решений стохастических дифференциальных уравнений $dX_t = f(X_t)dB_t$, $t\in[0,T]$ 
с многомерным дробным броуновским движением $B_t = (B^{(0)}_t, B^{(1)}_t, \ldots, 
B^{(d)}_t)^\top$, компоненты $B^{(i)}_t$ которого имеют различные показатели Харста 
$H_i > 1/3$, $i \ge 1$, а также содержащих снос $dB^{(0)}_t = dt$. В настоящей главе
рассматриваются решения $X^x_t$ данных уравнений с постоянными начальными 
условиями $X_0 = x$, $x\in\mathbb{R}^n$.

Получены асимптотические разложения тейлоровского типа для функционалов 
$\textbf{P}_t g(x) = \E g(X^{x}_t)$ от решений указанных уравнений при малых значениях
времени $t\ge 0$. Указанные разложения порядка $N\in\mathbb{N}$ имеют место, если 
функции $f$, $g$ имеют непрерывные и ограниченные производные до порядков $N+2$ 
и $N+3$ соответственно (включительно). Также в случае, когда все индексы Харста 
удовлетворяют уловию $H_i \ge H^* > 1/2$, приведены явные формулы вычисления 
математических ожиданий повторных интегралов от дробных броуновских движений, 
фигурирующих в разложениях.

В случае, когда дифференциальные операторы $D^{(i)}_f$, $0\le i \le d$ коммутируют, 
получены уравнения в частных производных колмогоровского типа для функций 
$\varphi(t, x) = \E g(X^{x}_t)$ в предположении, что функции $f$, $g$ имеют непрерывные 
и ограниченные производные до порядков $d+2$ и $d+3$ соответственно (включительно).

Приведены примеры, иллюстрирующие применение указанных теорем.

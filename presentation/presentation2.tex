\section{Введение}

\begin{frame}
  \frametitle{Введение}

  \begin{block}{Стохастическое дифференциальное уравнение}
    $$dX(t,\omega) = f(t,\omega, X(t,\omega))dt + g(t, \omega, X(t, \omega))dB(t,\omega),$$
    где $B(t, \omega)$ --- стандартное или дробное броуновское движение. 
  \end{block}

  \begin{block}{Интегральный критерий}
    Траектории процесса $B(t)$ недифференцируемы, 
    поэтому уравнение понимают в интегральном смысле:
    $$
      X(t,\omega) = X(s,\omega) + \int_s^t f(\tau,\omega, X(\tau,\omega))d\tau + 
      \int_s^t g(\tau, \omega, X(\tau, \omega))dB(\tau,\omega).
    $$
  \end{block}

  \begin{block}{Интегралы по дробному броуновскому движению}
    \begin{itemize}
      \item Потраекторные интегралы (Янга, Губинелли).
      \item Стохастические интегралы (Ито, Стратоновича, $\theta$- и $\mu$-интегралы) 
        для стандартного броуновского движения.
      \item Интегралы Вика-Ито-Скорохода.
    \end{itemize}
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Введение: теория грубых траекторий}

  \begin{block}{Одномерное ДУ $dy_t = f(y_t)dx_t$, управляемое $x_t\in C^{1-var}$}
    $$
      y_t = y_s + \int_s^t f(y_\tau)dx_\tau \Rightarrow
      y_t = y_s + \sum_{n=1}^\infty (V^n_f f)(y_s) \int_{s<t_1<\ldots < t_n < t}dx_{t_1}\ldots dx_{t_n},
    $$
    где $(V_f g)(x) = f(x)g'(x)$. Таким образом, можно положить по определению
    $$
      \int_s^t f(y_\tau)dx_\tau := \sum_{n=1}^\infty (V^n_f f)(y_s) 
      \int_{s<t_1<\ldots < t_n < t}dx_{t_1}\ldots dx_{t_n}.
    $$
    и поставить в соответствие $x\in C^{1-var} \mapsto \mathcal{J}(x) = 1 + \sum_{k=1}^\infty \int_{\Delta^k} dx^{\otimes k}$.
  \end{block}

  \begin{block}{Многомерное ДУ $dy_t = f(y_t)dx_t$, управляемое $x_t\in C^{p-var}$}
    \textbf{Теорема Лайонса} (1998). $x\in C^{p-var}(\mathbb{R}, \mathbb{R}^d) 
      \Rightarrow \mathcal{J}(x) = 1 + \sum_{k=1}^{[p]} \int_{\Delta^k} dx^{\otimes k}$,
      $\mathcal{J}(x) \in \mathbb{T}^{[p]}(\mathbb{R}^d) := 
      \otimes^{[p]}_{k=0} (\mathbb{R}^d)^{\otimes k}$.
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Введение: теория грубых траекторий}

  \begin{block}{Процесс второго порядка}
    Пусть $W$ --- конечномерное БП над $\mathbb{R}$. 
    Процессом второго порядка над $X\colon [0, T]\to W$ называется отображение
    $\mathbb{X}\colon[0,T]^2\to W^{\otimes 2}$, удовлетворяющее тождеству Чена:
    $$
      \mathbb{X}_{s,t} - \mathbb{X}_{s, u} - \mathbb{X}_{u, t} = 
      (X_u - X_s) \otimes (X_t - X_u), \forall (s,u,t)\in[0,T]^3.
    $$
  \end{block}

  \begin{block}{}
    \textbf{Пример}. Если отображение $X\colon [0, T]\to \mathbb{R}$ достаточно 
    регулярное и определен интеграл Янга
    $$
      I_{s,t} = \int_s^t (X_r - X_s)dX_r = \iint_{s<r<\tau<t} dX_r dX_\tau,
    $$
    то $I_{s,t}$ является примером процесса второго порядка над $X$, а тождество
    Чена выражает свойство аддитивности кратного интеграла:
    $$
      \int_s^t (X_r - X_s)dX_r = \int_s^u (X_r - X_s)dX_r + \int_u^t (X_r - X_u) dX_r + 
      (X_u - X_s)(X_t - X_u).
    $$
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 2: объект исследования}

  \begin{block}{Обозначения}
    Рассмотрим вероятностное пространство $(\Omega, \mathcal{F}, \Prob)$, на 
    котором определены независимые одномерные дробные броуновские 
    движения~$B_t^{(1)},\ldots,B_t^{(d)}$ с индексами Харста 
    $H_1,\ldots, H_d \in (1/3,1).$ 

    Введем обозначение 
    $B_t=(B_t^{(0)}, \ldots, B_t^{(d)})^\top$ для $(d+1)$-мерного дробного 
    броуновского движения, в котором $B_t^{(0)}=t.$ Пусть также $H_0=1.$
    Пусть $H_{\min}$ --- значение наименьшего из индексов Харста $H_i,$ 
    $i=0,\ldots, d.$ Выберем и зафиксируем некоторое $H \in (1/3, 1/2]$ такое, 
    что $H<H_{\min}$.
  \end{block}

  \begin{block}{Объект исследования}
    Cтохастическое дифференциальное уравнение
    \begin{equation}\label{chapter3_main_eq_compact}
      dX_t = f(X_t)dB_t, \ t \in [0,T],
    \end{equation}
    в котором $f$ --- $(n\times(d+1))$-матрица, столбцами которой являются векторы 
    $f_i:\mathbb{R}^n \to \mathbb{R}^n,$ $i=0,\ldots, d$. 
  \end{block}
  
\end{frame}

\begin{frame}{Глава 2: процесс второго порядка}

\begin{block}{Процесс второго порядка над $B_t$}
\vspace{-2ex}
\begin{gather*}
\mathbb{B}\colon [0,T]^2\times \Omega \to \mathbb{R}^{(d+1)\times(d+1)}, \quad
\mathbb{B}_{s,t} = \left(\mathbb{B}^{(i,j)}_{s,t}\right)^{d}_{i,j=0}.\\
\mathbb{B}^{(i,j)}_{s,t} \stackrel{L^2}{=}
\lim\limits_{|\mathcal{P}|\to 0}\int_{\mathcal{P}} B^{(i)}_{s,r} dB^{(j)}_{r}, 
\;\; \int_{\mathcal{P}} B^{(i)}_{s,r} dB^{(j)}_{r} \!=\!\!\!\!\! 
\sum\limits_{t_k, t_{k+1}\in\mathcal{P}} 
\!\!\!\!\!B^{(i)}_{s, t_k}B^{(j)}_{t_k, t_{k+1}},\quad 1\le i < j \le d,\\
\mathbb{B}^{(0,j)}_{s,t} =\int_s^t B^{(j)}_{s,r} dr \stackrel{\text{п.н.}}{=} 
\lim\limits_{|\mathcal{P}|\to 0} \sum\limits_{t_k, t_{k+1}\in\mathcal{P}} 
B^{(j)}_{s, t_k}(t_{k+1} - t_k),\quad 1\le j\le d,\\
\mathbb{B}^{(i,i)}_{s,t} = \frac{1}{2} \left(B^{(i)}_{s,t}\right)^2,
\quad 0\le i\le d,\\
\mathbb{B}^{(i,j)}_{s,t} = -\mathbb{B}^{(j,i)}_{s,t} + 
B^{(i)}_{s,t}B^{(j)}_{s,t},\quad 0\le j < i \le d
\end{gather*}
для любой пары $(s,t)\in [0,T]^2$, где 
$\mathcal{P} = \{s=t_0 < t_1 < \ldots < t_{l} =t \}$ --- 
произвольное разбиение отрезка $[s,t]$, $|\mathcal{P}| = 
\max |t_{k+1}-t_k|$, а все пределы понимаются не зависящими от 
последовательности разбиений $\mathcal{P}$. 

Обозначения 
$\stackrel{L^2}{=}$,  $\stackrel{\text{п.н.}}{=}$ применяются для того, 
чтобы показать, что соответствующие пределы понимаются в смысле 
$L^2(\Omega, \mathcal{F}, \Prob)$ и п.н. соответственно.
\end{block}

\end{frame}

\begin{frame}{Глава 2: элементы теории грубых траекторий}

\begin{block}{Обозначения}
Пусть $\alpha \in (1/3,1/2]$.

$V$, $W$ --- конечномерные банаховы пространства над $\mathbb{R}$.

$C^\alpha([0,T], W) = \left\{Z\colon[0;T]\to W\;\Bigl|\;\|Z\|_\alpha = \sup\limits_{s,t\in[0,T]: s\ne t}\frac{|Z_t - Z_s|_W}{|t-s|^\alpha} < \infty\right\}.$

Для функций $R \colon [0,T]^2 \to W$ обозначим $\|R\|_{2\alpha} = \sup_{s,t\in[0,T]: s\ne t}\frac{|R_{s,t}|_W}{|t-s|^{2\alpha}}.$

Для функций $Z \colon [0,T] \to W$ обозначим их приращения как $Z_{s,t} = Z_t - Z_s$.

$\mathcal{L}(W, V)$ --- пространство линейных ограниченных операторов, действующих из $W$ в~$V$.
\end{block}

\end{frame}

\begin{frame}{Глава 2: элементы теории грубых траекторий}

\begin{block}{Производная Губинелли}
Говорят, что функция $Y\in C^\alpha([0,T], \mathcal{L}(W, V))$ управляется 
функцией~$Z\in\linebreak\in C^\alpha([0,T], W)$, если существует $Y'\in C^\alpha\bigl([0,T], 
\mathcal{L}(W, \mathcal{L}(W, V))\bigr)$ (называемое производной Губинелли $Y$),
такое, что остаток $R^Y_{s,t} = Y_{s,t} - Y'_s Z_{s,t}$ удовлетворяет 
неравенству $\|R^Y\|_{2\alpha} < +\infty$. Множество всех $(Y, Y')$ таких, 
что~$Y$ управляется~$Z$, будем обозначать $\mathcal{D}^{2\alpha}_Z([0,T],
\mathcal{L}(W, V))$.
\end{block}

\begin{block}{}
\textbf{Пример}: $Z_t = t$, $Y_t - Y_s = \frac{dY_t}{dt}(t-s) + o(|t-s|)$. 
Производная Губинелли $Y' = \frac{dY_t}{dt}$
\end{block}

\begin{block}{Множество управляемых функций}
В терминах предыдущего определения, множество всех $(Y, Y')$ таких, 
что~$Y$ управляется~$Z$, будем обозначать $\mathcal{D}^{2\alpha}_Z([0,T],
\mathcal{L}(W, V))$.
\end{block}

\end{frame}

\begin{frame}{Глава 2: потраекторный интеграл Губинелли и определение решения}

\begin{block}{Потраекторный интеграл Губинелли (П. Фриц, М. Хайрер, 2014)}
Пусть $(Y, Y')\in \mathcal{D}^{2\alpha}_Z([0, T], \mathcal{L}(W, V))$. Потраекторным 
интегралом Губинелли $Y$ по $Z$ называют предел интегральных сумм
$$
\int_0^T Y_t \,dZ_t = \lim_{|\mathcal{P}|\to 0}\sum_{t_i, t_{i+1}\in\mathcal{P}}
(Y_{t_i}Z_{t_i, t_{i+1}} + Y'_{t_i}\mathbb{Z}_{t_i, t_{i+1}}),
$$
где $|\mathcal{P}| = \max |t_{i+1}{-}t_i|$ --- диаметр разбиения 
$\mathcal{P} = \{0{=}t_0 {<} t_1 {<} \ldots {<} t_{l} {=}T \}$, а предел понимается не 
зависящим от последовательности разбиений $\mathcal{P}$.
\end{block}

\begin{block}{Решение основного уравнения}
Случайный процесс $X_t$ такой, что $(X, X')\in
\mathcal{D}^{2H}_B([0, T], \mathbb{R}^n)$ п.н., будем называть решением 
уравнения~(\ref{chapter3_main_eq_compact}), если он п.н. удовлетворяет равенству
$$
X_t = X_0 + \int_0^t f(X_s) dB_s,\quad t\in [0, T],
$$
где интеграл понимается как потраекторный интеграл Губинелли.
\end{block}

\end{frame}

    Через $C^k_b(V, U)$ будем обозначать пространство функций

    ограниченные производные до порядка $k$ включительно с нормой

  \end{block}

  \begin{block}{Теорема 2.1.}
    Если $f\in C^2_b(\mathbb{R}^n, \mathbb{R}^{n\times(d+1)})$, то для

    всех $i=0,\ldots, d,$ то справедливо включение 
    $X \in C^{H^*}([0,T], \mathbb{R}^{n})$ п.н. и интеграл в определении решения 
    уравнения \textup{(\ref{chapter3_main_eq_compact})} является потраекторным
    интегралом Янга.
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Глава 2: основные результаты}

  \begin{block}{Теорема 2.2. [5].}
    Пусть $f\in C^3_b(\mathbb{R}^n, \mathbb{R}^{n \times (d+1)})$, 
    $g\in C^3_b(\mathbb{R}^n, \mathbb{R})$. Тогда для любых $s,$ $t \in [0,T]$ п.н. 
    справедлива следующая формула замены переменных:
    \begin{equation}
      g(X_t) = g(X_s) + \int_s^t Dg(X_r)f(X_r)dB_r,\qquad s,t\in [0, T],
    \end{equation}
    где $X_t$ --- решение уравнения \textup{(\ref{chapter3_main_eq_compact})}.
  \end{block}

  \begin{block}{}
    Наряду с уравнением~(\ref{chapter3_main_eq_compact}) рассмотрим аналогичное 
    уравнение с возмущенной правой частью
    \begin{equation}\label{stability_intro_main_eq_1}
      d\widetilde X_t = \widetilde f(\widetilde X_t)dB_t,\quad t\in[0,T],
    \end{equation}
    в котором $\widetilde f$ --- $(n\times(d+1))$-матрица, столбцами которой являются 
    векторы $\widetilde f_i\colon\mathbb{R}^n \to \mathbb{R}^n,$ $i=0,\ldots,d$.
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 2: основные результаты}

  \begin{block}{Теорема 2.3. [3].}
    Пусть $f,\widetilde f \in C^3_b(\mathbb{R}^n, \mathbb{R}^{n\times(d+1)})$, 
    причем функция $\widetilde f$ такова, что $\|f - \widetilde f\|_{C^2_b} \le 1$. 
    Обозначим через $X_t$, $\widetilde X_t$ решения 
    уравнений~\textup{(\ref{chapter3_main_eq_compact})}, 
    \textup{(\ref{stability_intro_main_eq_1})} c начальными условиями 
    $X_0=\xi$, $\widetilde X_0 = \widetilde \xi$ соответственно. Тогда:

    1) почти наверное справедлива следующая оценка
    \begin{equation}
      \|X - \widetilde X\|_H \le C \left(|\xi - \widetilde \xi| + 
      \|f - \widetilde f\|_{C^2_b}\right)
    \end{equation}
    для некоторой случайной величины $C=C(H, T, \|f\|_{C^3_b}, \|B\|_H, 
    \|\mathbb{B}\|_{2H})$. Причем~$C$ может быть выбрана не зависящей от~$T$, 
    если $T\in (0,1];$

    2) имеет место следующее неравенство
    \begin{equation}
      \mathbb{E}\bigl(\ln\|X - \widetilde X\|_H\bigr) \le C + 
      \ln\bigl(\mathbb{E}|\xi - \widetilde\xi| + \|f-\widetilde f\|_{C^2_b}\bigr),
    \end{equation}
    где $C = C(H, H_1, \ldots, H_d, T, \|f\|_{C^3_b})\in\mathbb{R}$ --- 
    константа, вообще говоря, зависящая от $H$, $H_1,\ldots, H_d$, $T$, 
    $\|f\|_{C^3_b}$.
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 3: основной результат}

  \begin{block}{Теорема 3.1. [4, 5].}
    Пусть $f\in C^{N+2}_b(\mathbb{R}^n, \mathbb{R}^{n \times (d+1)})$,
    $g\in C^{N+3}_b(\mathbb{R}^n, \mathbb{R})$, $N\in\mathbb{N}$. Тогда для любого 
    фиксированного $H\in(1/3, 1/2]$ такого, что $H < H_{min} = 
    \min\limits_{i=0,\ldots d} H_i$ справедливо следующее асимптотическое разложение:
    \begin{equation}
      \textup{\textbf{P}}_t g(x) = g(x) + 
      \sum_{k=1}^N \sum_{I_{k}\in\{0,\ldots, d\}^{k}} t^{|H_{I_k}|}\cdot 
      (D^{(I_{k})}_f g)(x)\E\biggl(\int_{\Delta^{k}[0,1]}dB^{(I_{k})}\biggr) + 
      O(t^{(N+1)H}),
    \end{equation}
    при $t\to 0$, где $|H_{I_k}| = H_{i_1} + H_{i_2}+ \ldots + H_{i_k}$ --- сумма 
    индексов Харста дробных броуновских движений $B^{(i_1)}, B^{(i_2)},\ldots, B^{(i_k)}$.
  \end{block}

\begin{frame}

  \begin{block}{Простейшее уравнение}

    считая, что $d=1$. Вместе с ним рассмотрим простейшее уравнение
    \begin{equation}\label{integration_of_mixed_simplest_eq}
      dy(t)=u(t)dt + v(t)dW(t) + b(t)dB(t),\quad t\ge 0.
    \end{equation}
  \end{block}

  \begin{block}{Решение простейшего уравнения}
    Решение уравнения~(\ref{integration_of_mixed_simplest_eq}) определяется следующим 
    соотношением:
    \begin{equation*}
      y(t) = y(0) + \int_0^t u(\tau)d\tau + \int_0^t v(\tau)dW(\tau) + 
      \int_0^t b(\tau)dB(\tau),\quad t\ge 0.
    \end{equation*}
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 4: основные результаты}

    помощью некоторого дважды непрерывно дифференцируемого и обратимого 
    относительно $x$ преобразования $y=F(t, x)$ тогда и только тогда, когда 
    найдутся функции $q(t), r(t)$ такие, что оказываются выполненными

  \end{block}

\begin{frame}
  \frametitle{Глава 4: основные результаты}

  \begin{block}{Уравнение Стратоновича}
    Вернемся к уравнению~(\ref{integration_of_mixed_main_eq}) в пространстве 
    $\mathbb{R}^d$ и вместе с ним рассмотрим соответствующее уравнение Стратоновича
    \begin{equation}\label{eq_strat}
      dx(t) = (f(t,x(t))-c(t,x(t)))dt + g(t, x(t))\circ dW(t) + \sigma(t, x(t)) dB(t),
      \quad t \ge 0,
    \end{equation}
    в котором $f\colon\mathbb{R}^+\times\mathbb{R}^d\to\mathbb{R}^d$,
    $g\colon\mathbb{R}^+\times\mathbb{R}^d\to\mathbb{R}^{d \times d}$,
    $\sigma\colon\mathbb{R}^+\times\mathbb{R}^d\to\mathbb{R}^{d \times d}$,
    $c\colon \mathbb{R}^+\times\mathbb{R}^d\to\mathbb{R}^d$ и
    $$
      c_i(t,x)=\frac{1}{2}\sum_{j,k=1}^d \frac{\partial g_{ij}(t,x)}{\partial x_k}g_{kj}(t,x),
      \quad i=1, \ldots, d.
    $$
  \end{block}

  \begin{block}{Теорема 4.3. [6].}
    Пусть функция $f$ непрерывна, функция $g$ непрерывна вместе со своими 
    производными $\frac{\partial g}{\partial t}$, $\frac{\partial g}{\partial x}$,
    $\frac{\partial^2 g}{\partial x^2}$, а функция~$\sigma$ удовлетворяет
    $(\alpha,\beta)$-условию Гельдера по $(t,x)$ для некоторых $\alpha>1-H,$ 
    $\beta>2-2H$. Процесс $x(t)$ является решением уравнения 
    \textup{(\ref{integration_of_mixed_main_eq})} тогда и только тогда, когда 
    процесс $x(t)$ является решением уравнения Стратоновича~\textup{(\ref{eq_strat})}.
  \end{block}

\begin{frame}
  \frametitle{Глава 5, раздел 5.2: объект исследования}

  \vspace{-1ex}
  \begin{block}{Объект исследования}
    Стохастическое дифференциальное уравнение в  
    пространстве $\mathbb{R}^d$ c выделенной линейной частью вида:
    \begin{equation}\label{mainSys}
      dX(t) = (A(t)X(t) + f(t, X(t)))dt + g(t, X(t))dW(t),\quad t\ge 0.
    \end{equation}
    где $W(t)$ --- стандартное $d$-мерное броуновское движение,
    $A\colon\mathbb{R}^+\to\mathbb{R}^{d\times d}$~--- кусочно 
    непрерывная функция, $\sup_{t\ge 0}|A(t)|\le~M$,
    $f\colon \mathbb{R}^{+} {\times} \mathbb{R}^{d} {\to}\mathbb{R}^{d}$, 
    $g\colon \mathbb{R}^{+} \times \mathbb{R}^{d} \to 
    \mathbb{R}^{d \times d}$ --- измеримые по Борелю функции такие, что:
    
    1) $f(t, 0) = 0$ и $g(t, 0) = 0$ при всех $t\in\mathbb{R}^+$ 

    2) существует постоянная~$C$ такая, что для любых $t \in \mathbb{R}^{+}$, $x \in 
    \mathbb{R}^{d}$, выполняется неравенство 
    $|f(t, x)| + |g(t, x)| \le C(1+|x|)$ (выполнено условие линейного порядка роста по $x$).
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.2: основной результат}

  \vspace{-1ex}
  \begin{block}{Детерминированное линейное приближение}
    \begin{equation}\label{auxillarySys}
      dX(t) = A(t)X(t)dt,\quad t\ge 0,
    \end{equation}
    Обозначим через $X^{(s,x)}(t)$ решение уравнения~(\ref{auxillarySys}), 
    такое, что $X^{(s,x)}(s) = x$.
  \end{block}

  \vspace{-1ex}
  \begin{block}{}
    Будем говорить, что уравнение~(\ref{auxillarySys}) имеет 
    \textit{равномерно экспоненциально устойчивое} нулевое решение, если существуют 
    константы $\Lambda, \lambda > 0$, не зависящие от $s$, $x$, такие, что для любых 
    $s\in\mathbb{R}^+$, $x\in\mathbb{R}^d$ и~$t\ge s$ выполняется неравенство 
    $
      |X^{(s,x)}(t)|^2\le\Lambda|x|^2 e^{-\lambda(t-s)}.
    $
  \end{block}

  \vspace{-1ex}
  \begin{block}{Теорема 5.3. [1].}
    Предположим, что функции $f(t, x)$ и $g(t, x)$ таковы, что выполнены соотношения
    \begin{equation}
      \frac{|f(t, x)|}{|x|}\xrightarrow[x\to 0]{} 0,\qquad \frac{|g(t, x)|}{|x|}\xrightarrow[x\to 0]{} 0
    \end{equation}
    равномерно по $t\in\mathbb{R}^+$,  
    а система~\textup{(\ref{auxillarySys})} имеет равномерно экспоненциально 
    устойчивое нулевое решение. Тогда система~\textup{(\ref{mainSys})} имеет 
    асимптотически устойчивое по вероятности  нулевое решение.
  \end{block}

\end{frame}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: объект исследования}

  \begin{block}{Обозначения}
    Пусть $H$ и $K$ --- сепарабельные гильбертовы пространства, \\ \smallskip
    $\mathfrak{L}_2(K,H)$ --- пространство операторов Гильберта-Шмидта, 
    действующих из $K$ в $H$,\\ \smallskip
    $Q$~--- ядерный симметрический положительно 
    определенный оператор на пространстве~$K$, \\ \smallskip
    $W(t,\omega)$ --- $Q$-броуновское 
    движение со значениями в $K$ и ковариационным оператором $Q$.
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: объект исследования}

  \begin{block}{Объект исследования}
    Эволюционное функциональное уравнение в 
    пространстве $H$ следующего вида:
    \begin{equation}\label{MostGeneralEq}
      dX(t,\omega)=AX(t,\omega)dt+f(t, X(t, \omega))dt+
      g(t, X(t, \omega))dW(t,\omega),
      \ (t,\omega) \in \mathbb{R}^+\times\Omega
    \end{equation}
    относительно $X \in H$ с начальным условием
    \begin{equation}\label{MostGeneralStartCond}
      X(0,\omega)=\xi(\omega),  \quad \omega \in \Omega,
    \end{equation}
    где $f\colon\mathbb{R}^+\times H \to H$,
    $g\colon\mathbb{R}^+\times H \to \mathfrak{L}_2(K,H)$ --- 
    измеримые, непрерывные по $X$ (при любом фиксированном 
    $t\in\mathbb{R}^+$) функции,\\ \smallskip
    $A$ --- линейный оператор, определенный на всюду плотном в $H$ 
    множестве $\EuScript D(A)$ и порождающий $C_0$-полугруппу 
    $S(t)$ на~$H$, \\ \smallskip
    $\xi:\Omega \to \EuScript D(A)$ --- 
    $\mathcal{F}_0$-измеримая случайная величина, имеющая конечный 
    момент $\E\|\xi\|^p<\infty$ порядка $p>2$.
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: объект исследования}

  \begin{block}{Условия на коэффициенты $f$, $g$}
    Относительно функций $f(t, X)$ и $g(t, X)$ будем 
    предполагать, что выполнены два условия:

    \begin{enumerate}
      \item\textit{Локальное условие Липшица}. Для любого $a > 0$ существует 
        постоянная~$q_a$ такая, что для всех $t\in[0,a]$ и любых 
        $\,\varphi,\psi\in H$, таких, что
        $\|\varphi\|\le a$, $\|\psi\|\le a\,$, выполняются неравенства
        $$
          \|f(t, \varphi) - f(t, \psi)\|\le q_a\|\varphi-\psi\|,\quad
          \|g(t,\varphi) - g(t,\psi)\|\le q_a\|\varphi-\psi\|.
        $$
      \item \textit{Условие линейного порядка роста}. Существует непрерывная 
        функция~$k\colon\mathbb{R}^+\to\mathbb{R}^+$ такая, что для всех 
        $t\in\mathbb{R}^+$ и любого $\eta\in H$ выполняются неравенства
        $$
          \|f(t,\eta)\|\le k(t)(1+ \|\eta\|), \quad \|g(t,\eta)\|\le k(t)(1+ \|\eta\|).
        $$
    \end{enumerate}
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: притяжение к нулю}

  \begin{block}{О\,п\,р\,е\,д\,е\,л\,е\,н\,и\,е\, 5.8.}
    Пусть положительная функция $\lambda(t)$ 
    определена для достаточно больших $t>0$, скажем, $t\ge T > 0$. Предположим, что
    \begin{enumerate}
      \item $\lim\limits_{t\to\infty}\lambda(t) = \infty$.
      \item $\ln\lambda(t)$ равномерно непрерывна по $t\ge T$.
      \item Существует константа $\tau\ge 0$ такая, что 
      $\lim_{t\to\infty}\sup\frac{\ln\ln t}{\ln\lambda(t)}\le\tau$.
    \end{enumerate}
    Будем говорить, что слабое решение задачи~(\ref{MostGeneralEq}),~(\ref{MostGeneralStartCond})
    притягивается к нулю со скоростью $\lambda(t)$, если найдется $\gamma>0$ такое, 
    что п.н. выполняется неравенство
    $$
      \lim_{t\to\infty}\sup\frac{\ln\|X(t)\|}{\ln\lambda(t)}\le-\gamma.
    $$
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: основной результат}

  \begin{block}{Обозначения}
    Введем операторы $L$, $B$, действующие на функционал 
    $V(t,x)\in C^{1,2}(\mathbb{R}^+{\times} H, \mathbb{R}^+)$ 
    следующим образом:
    \begin{gather}
      LV(t,x) = V'_t(t,x) + \langle V'_x(t,x), Ax + f(t, x)\rangle_H + 
      \nonumber\\ 
      \label{3.1}
      +\frac{1}{2}\textup{tr}[V''_{xx}(t,x)(g(t, x)Q^{1/2})
      (g(t, x)Q^{1/2})^\star],
      \quad (t,x)\in\mathbb{R}^+{\times} \EuScript D(A),\\
      BV(t,x) = \textup{tr}[V''_{xx}(t,x)\otimes 
      V''_{xx}(t,x)(g(t, x)Q^{1/2})(g(t, x)Q^{1/2})^\star],\nonumber\\
      \label{3.2}
      (t,x)\in\mathbb{R}^+{\times}H.
    \end{gather}
    ($Q$ --- ковариационный оператор броуновского движения $W(t)$)
  \end{block}
\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: основной результат}

  \begin{block}{Теорема 5.4. [2].}
    Пусть задан функционал $V(t,x)\in C^{1,2}(\mathbb{R}^+{\times} H, 
    \mathbb{R}^+)$ и две неотрицательные непрерывные функции 
    $\psi_1(t)$, $\psi_2(t)$. Предположим, что существуют 
    постоянные $r>0$, $m\ge0$, постоянные~$\mu,\nu,\theta\in\mathbb{R}$ и 
    невозрастающая положительная функция $\zeta(t)$ такие, что 
    $\frac{m-\left(\max\{\nu,\mu+\tau\}+\theta\right)}{r}> 0 $ и выполнены 
    следующие условия:
    \begin{enumerate}
      \item $\|x\|^{r}(\lambda(t))^m \le V(t,x)$ для всех $(t,x)\in 
        \mathbb{R}^+{\times}H$.
      \item $LV(t, x) + \zeta(t)BV(t, x) \le \psi_1(t) + 
        \psi_2(t)V(t, x)\;$ для всех $(t,x)\in \mathbb{R}^+{\times}\EuScript D(A)$.
      \item $\lim\limits_{t\to+\infty}\sup\frac{
        \ln\left(\int_0^t\psi_1(s)\,ds\right)}{\ln\lambda(t)}\,{\le}\,\nu$, 
        $\quad\lim\limits_{t\to+\infty}\sup
        \frac{\int_0^t\psi_2(s)\,ds}{\ln\lambda(t)}\,{\le}\,\theta$, 
        $\quad\lim\limits_{t\to+\infty}\inf\frac{\ln\zeta(t)}{\ln\lambda(t)}\,
        {\ge}\,{-}\mu$.
    \end{enumerate}
    Тогда слабое решение 
    задачи~\textup{(\ref{MostGeneralEq})},~\textup{(\ref{MostGeneralStartCond})} 
    притягивается к нулю со скоростью~$\lambda(t)$.
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: пример}

  \vspace{-1ex}
  \begin{block}{}
    \vspace{-2ex}
    \begin{gather*}
      dX_t(x) = \left(\frac{d^2}{dx^2}X_t(x) + \alpha \sin\left(X_t(x)+
      e^{-\frac{mt}{2}}\cos X^1_t\right)\right)dt + 
      \alpha e^{-\frac{mt}{2}} X_t(x) dW_t,\\
      dX^1_t = \left(\alpha X^1_t\sin X^1_t + 
      \left(\int_0^\pi X_t(x)^2dx\right)^{1/2}\right)dt + 
      \alpha e^{-\frac{mt}{2}}\left(\int_0^\pi X_t(x)^2dx\right)^{1/2} \!\!\!dW_t.
    \end{gather*}
  \end{block}
  \begin{block}{}
    Уравнение относительно $\bar X_t = (X_t(\cdot),X^1_t)^\top$, $t>0$, в пространстве 
    $H\times\mathbb{R}$ с начальным условием $\bar X_0=(X_0(x), X^1_0)^\top = (x_0(x), x^1_0),
    x\in (0,\pi)$, $K = \mathbb{R}$, $H = L_2[0,\pi]$ в компактной форме примет вид
    \begin{gather}\label{3.18}
      d\bar X_t = (\bar A\bar X_t + f(t, \bar X_t))dt + g(t, \bar X_t)dW_t,\\ \nonumber
      f(t, \bar X_t) = \alpha\left(\sin(X_t(x)+e^{-\frac{mt}{2}}\cos X^1_t), 
      X^1_t \sin X^1_t + \|X_t(x)\|_H\right)^\top,\\ \nonumber
      g(t, \bar X_t) = \alpha e^{-\frac{mt}{2}}\left(X_t(x), 
      \left(\int_0^\pi X_t(x)^2dx\right)^{1/2}\right)^\top, \ 
      \bar A = \begin{pmatrix}A & 0\\ 0 & 0\end{pmatrix}, \\ \nonumber
      A = \frac{d^2}{dx^2}, \quad \EuScript{D}(A) = \{u\in C_2[0,\pi]: 
      u(0)=u(\pi) = 0\}
    \end{gather}
  \end{block}

\end{frame}

\begin{frame}
  \frametitle{Глава 5, раздел 5.3: пример}

  \begin{block}{Применение теоремы 5.4}
    Положим $V(t,\bar u) = V(t, u) = e^{mt} \|u\|^2,\, u\in H$, $\lambda(t)=e^t$, 
    $\psi_1(t)\equiv \alpha\pi$, $\psi_2(t)\equiv\theta$, $\tau=\mu=\nu=0$. 

    В статье [2] показано, что при таком выборе, достаточно 
    малом $\alpha$ и достаточно большом $m$, условия 
    теоремы~5.4 будут выполнены, т.е. имеет место притяжение 
    решений к нулю.

    При этом функция~$g$ удовлетворяет глобальному условию 
    Липшица, а функция~$f$ удовлетворяет локальному, но не удовлетворяет
    глобальному условию Липшица.
  \end{block}

\end{frame}

  \end{enumerate}

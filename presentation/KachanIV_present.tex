\begin{frame}{Введение}

\begin{block}{}
Представленная магистерская диссертация посвящена доказательству теорем об устойчивости решений стохастических дифференциальных уравнений и исследованию асимптотических свойств решений при малых значениях времени.
\end{block}

\begin{block}{}
Основная часть представленной диссертации включает в себя 4 главы. В первой главе работы приводятся некоторые сведения из теорий случайных процессов, в частности дробных броуновских движений и интегралов по ним, функционального анализа, дифференциальных уравнений и др. Вторая глава работы посвящена исследованию стохастических дифференциально-функциональных уравнений со стандартным броуновским движением в гильбертовых пространствах. Третья глава работы посвящена развитию асимптотической теории стохастических дифференциальных уравнений, управляемых дробными броуновскими движениями с различными показателями Харста, большими 1/3.  В четвертой главе работы доказана теорема о непрерывной зависимости от начальных данных решений стохастических дифференциальных уравнений с дробными броуновскими движениями.
\end{block}

\end{frame}



\begin{frame}{Глава 2, раздел 1. Объект исследования и основной результат.}

\begin{block}{Нелинейное СДУ в $\mathbb{R}^d$ с выделенной линейной частью и запаздыванием}
\begin{equation}\label{mainSys}
dX(t) = (A(t)X(t) + f(t, X(t), X_t))dt + g(t, X(t), X_t)dW(t),\quad t\ge 0,
\end{equation}
где  $f : \mathbb{R}^{+} \times \mathbb{R}^{d} \times C_h \to\mathbb{R}^{d}$, $g : \mathbb{R}^{+} \times \mathbb{R}^{d} \times C_h \to \mathbb{R}^{d \times d}$ -- измеримые по Борелю функции, имеющие линейный порядок роста, $X_t = \{X(t+\tau):\, -h\le\tau\le 0\}\in C_h$, $C_h = C\left([-h,0], \mathbb{R}^d\right)$, $h>0$ --- время запаздывания, $A\colon\mathbb{R}^+\to\mathbb{R}^{d\times d}$ --- кусочно-непрерывная функция такая, что $\sup_{t\ge 0}|A(t)|\le M$, $W(t)$ -- $d$-мерное броуновское движение.
\end{block}

\begin{block}{Линеаризованное однородное детерминированное уравнение}
\begin{equation}\label{auxillarySys}
dX(t) = A(t)X(t)dt,\quad t\ge 0,
\end{equation}
\end{block}

\begin{block}{Теорема 2.3 (об устойчивости по линейному приближению)}
Предположим, что функции $f(t, x, \varphi)$ и $g(t, x, \varphi)$ таковы, что при достаточно малом $\varepsilon > 0$ найдется $\delta_\varepsilon > 0$ такое, что выполняются  неравенства
$$
|f(t, x, \varphi)|\le\varepsilon|x|,\qquad |g(t, x, \varphi)|\le\varepsilon|x|,
$$
для любых $\,x$, $|x|\le \delta_\varepsilon,\,t\in\mathbb{R}^+,\,\varphi\in C\left([-h,0], \mathbb{R}^d\right)$, а уравнение \textup{(\ref{auxillarySys})} имеет равномерно экспоненциально устойчивое нулевое решение. Тогда уравнение~\textup{(\ref{mainSys})} имеет асимптотически устойчивое по вероятности  нулевое решение.  
\end{block}

\end{frame}

\begin{frame}{Глава 2, раздел 1. Пример 2.1}

\begin{block}{}
Приведем пример уравнения, имеющего асимптотически устойчивое по вероятности слабое нулевое решение на основании теоремы 2.3. Рассмотрим систему стохастических дифференциальных уравнений~(\ref{suffCond_eq_17})
\begin{gather}
dx(t)=((-20-0.1 \sin t)x(t)+0.1 \cos t \ y(t)+\sin^2 x(t))dt,\nonumber\\ \label{suffCond_eq_17}
dy(t)=(-0.1 \cos t \  x(t)-(20+0.1 \sin t)y(t))dt+\sin^2 y(t)
\textrm{sgn}(x(t{-}1))dw(t),
\end{gather}
при $t\ge 0$ с начальными условиями $x(t)=\psi(t),$ $t \in [-1,0],$ $y(0)=y_0,$
где $x,$ $y \in\mathbb{R},$ $w(t)$ -- одномерное броуновское движение. Нулевое решение линеаризованной системы 
\begin{gather*}
dx(t)=((-20-0.1 \sin t)x(t)+0.1 \cos t \ y(t))dt,\nonumber\\ 
dy(t)=(-0.1 \cos t \  x(t)-(20+0.1 \sin t)y(t))dt,
\end{gather*}
является равномерно экспоненциально устойчивым, и согласно теореме 2.3 нулевое решение системы~(\ref{suffCond_eq_17}) асимптотически устойчиво по вероятности.
\end{block}

\end{frame}


\begin{frame}{Глава 2, раздел 2. Описание объекта исследования.}

\begin{block}{Нелинейное СДУ в гильбертовом пространстве}
Пусть $H$~и~$U$ --- сепарабельные гильбертовы пространства, $W(t,\omega)$ --- броуновское движение со значениями в $U$, $C_h = C([-h,0], H)$.
\begin{equation}\label{MostGeneralEq}
dX(t,\omega)=AX(t,\omega)dt+f(t,\omega,X_t(\omega))dt+g(t,\omega, X_t(\omega))dW(t,\omega),
 \quad(t,\omega) \in \mathbb{R}^+\times\Omega
\end{equation}
\begin{equation}\label{MostGeneralStartCond}
X(t,\omega)=\xi(t,\omega),  \quad t\in[-h,0],\;\omega \in \Omega,
\end{equation}
где $X_t(\omega) = \{X(t{+}\tau,\,\omega)|\,\tau\in[-h,0]\}\in C_h$, $A$ --- линейный оператор, порождающий $C_0$-полугруппу $S(t)$ на~$H$, $\xi:[-h,0]\times\Omega \to \EuScript D(A)$ --- непрерывный случайный процесс, имеющий конечный момент $\E\sup_{t\in[-h,0]}\|\xi(t)\|^p<\infty$ порядка $p>2$, $f\colon\mathbb{R}^+\times\Omega\times C_h \to H$, $g\colon\mathbb{R}^+\times\Omega \times C_h \to L_2(U,H)$ --- функции, удовлетворяющие двум условиям:
\begin{enumerate}
\item \textbf{Локальное условие Липшица.} Для любого $a > 0$ существует постоянная~$q_a$ такая, что для всех $t\in[0,a]$ и любых $(\mathcal{F}, \mathcal{B}(C_h))$-измеримых случайных величин $\,\varphi,\psi\colon\Omega\to C_h$, таких, что п.н. $\|\varphi\|\le a$, $\|\psi\|\le a\,$:
$$
\|f(t,\omega,\varphi) - f(t,\omega,\psi)\|\le q_a\|\varphi-\psi\|,\quad \|g(t,\omega,\varphi) - g(t,\omega,\psi)\|\le q_a\|\varphi-\psi\|.
$$
\item \textbf{Условие линейного порядка роста.} Существует непрерывная функция~$k\colon\mathbb{R}^+\to\mathbb{R}^+$ такая, что для всех $t\in\mathbb{R}^+$ и любой $(\mathcal{F}, \mathcal{B}(C_h))$-измеримой случайной величины $\eta\colon\Omega\to C_h$ такой, что $\E\|\eta\|^p{<}\infty$:
$$
\|f(t,\omega,\eta)\|\le k(t)(1+ \|\eta\|), \quad \|g(t,\omega,\eta)\|\le k(t)(1+ \|\eta\|).
$$
\end{enumerate}
\end{block}

\end{frame}


\begin{frame}{Глава 2, раздел 2. Основной результат}

\begin{block}{Операторы ляпуновского типа}\vspace{-1em}
\begin{gather*}
LV(t,x,\varphi) = V'_t(t,x) + \langle V'_x(t,x), Ax + f(t,\varphi)\rangle + \nonumber\\
+\frac{1}{2}\textup{tr}[V''_{xx}(t,x)(g(t,\varphi)Q_w^{1/2})(g(t,\varphi)Q_w^{1/2})^\star],\quad (t,x,\varphi)\in\mathbb{R}^+{\times} \EuScript D(A){\times} C_h,\\
QV(t,x,\varphi) = \textup{tr}[V''_{xx}(t,x)\otimes V''_{xx}(t,x)(g(t,\varphi)Q_w^{1/2})(g(t,\varphi)Q_w^{1/2})^\star],\nonumber\\ 
(t,x,\varphi)\in\mathbb{R}^+{\times}H{\times}C_h.
\end{gather*}
\end{block}

\begin{block}{Теорема 2.4 (об асимптотической устойчивости)}
\label{HStabilityTheorem}
Пусть задан функционал $V(t,x)\in C^{1,2}(\mathbb{R}^+{\times} H, \mathbb{R}^+)$ и две неотрицательные непрерывные функции $\psi_1(t)$, $\psi_2(t)$. Предположим, что существуют положительные постоянные $r>0$, $m\ge0$, постоянные~$\mu,\nu,\theta\in\mathbb{R}$ и невозрастающая положительная функция $\zeta(t)$ такие, что $\frac{m-\left(\max\{\nu,\mu+\tau\}+\theta\right)}{r}> 0 $ и выполнены следующие условия:
\begin{enumerate}
\item $\|x\|^{r}(\lambda(t))^m \le V(t,x)$ для всех $(t,x)\in \mathbb{R}^+{\times}H$.
\item $LV(t, x,\varphi) + \zeta(t)QV(t, x, \varphi) \le \psi_1(t) + \psi_2(t)V(t, x)\;$ для всех $\;t\in\mathbb{R}^+$, $x\in\EuScript D(A)\;$ и $\;\varphi\in C_h$.
\item $\lim\limits_{t\to+\infty}\sup\frac{\ln\left(\int_0^t\psi_1(s)\,ds\right)}{\ln\lambda(t)}\,{\le}\,\nu$, $\quad\lim\limits_{t\to+\infty}\sup\frac{\int_0^t\psi_2(s)\,ds}{\ln\lambda(t)}\,{\le}\,\theta$, $\quad\lim\limits_{t\to+\infty}\inf\frac{\ln\zeta(t)}{\ln\lambda(t)}\,{\ge}\,{-}\mu$.
\end{enumerate}
Тогда слабое решение задачи~\textup{(\ref{MostGeneralEq})},~\textup{(\ref{MostGeneralStartCond})} асимптотически устойчиво.
\end{block}

\end{frame}


\begin{frame}{Глава 2, раздел 2. Пример 2.2}

\begin{block}{}
Рассмотрим следующую стохастическую дифференциальную систему
\begin{gather*}
dX_t(x) = \left(\frac{d^2}{dx^2}X_t(x) + \alpha \sin\left(X_t(x)+e^{-\frac{mt}{2}}\cos X^1_t\right)\right)dt + \alpha e^{-\frac{mt}{2}} X_t(x) dW_t,\\
dX^1_t = \left(\alpha X^1_t\sin X^1_t + \left(\int_0^\pi X_t(x)^2dx\right)^{1/2}\right)dt + \alpha e^{-\frac{mt}{2}}\left(\int_0^\pi X_t(x)^2dx\right)^{1/2}  dW_t,\\
 t>0,\;\; 0<x<\pi
\end{gather*}
как уравнение относительно $\bar X_t = (X_t(\cdot),X^1_t)^\top$ в пространстве $H\times\mathbb{R}$ 
с начальным условием $\bar X_0=(X_0(x), X^1_0)^\top = (x_0(x), x^1_0),\ x\in (0,\pi)$, $H = L_2[0,\pi]$, $U = \mathbb{R}$.\smallskip

\quad Положим $V(t,\bar u) = V(t, u) = e^{mt} \|u\|^2,\, u\in H$, $\zeta(t) \equiv 1$, $r=2$, $\psi_1(t)\equiv \alpha\pi$, $\psi_2(t)~\equiv~\beta$, $\lambda(t)=e^t$, $\tau = \mu = \nu = 0$, $\lambda_0=\inf_{u\in D(A)}\frac{\|u'\|^2}{\|u\|^2}\ge\frac{1}{\pi^2}>0$. Можно показать, что при указанном выборе, при достаточно больших $m>0$ и достаточно малых $\alpha > 0$ все условия теоремы 2.4 будут соблюдены и, следовательно,
$$\limsup_{t\to+\infty}\frac{\log\|X(t)\|}{t}\le-\frac{m-\left(\max\{\nu,\mu+\tau\}+\theta\right)}{r} = -\frac{2\lambda_0 - 3\alpha-10\alpha^2 }{2}<0.$$
Причем коэффициент $g$ уравнения удовлетворяет глобальному условию Липшица, а коэффициент $f$ удовлетворяет локальному, но не удовлетворяет глобальному условию Липшица.
\end{block}

\end{frame}


\begin{frame}{Главы 3, 4. Описание объекта исследования}

\begin{block}{}
Пусть $B_t^{(1)},\ldots,B_t^{(d)}$ --- независимые одномерные дробные броуновские движения с индексами Харста $H_1,\ldots, H_d \in (1/3,1).$ Обозначим $B_t=(B_t^{(0)}, \ldots, B_t^{(d)})^\top$ $(d+1)$-мерное дробное броуновское движение, в котором $B_t^{(0)}=t$, $H_0=1.$
Пусть $H_{\min}$ --- значение наименьшего из индексов Харста $H_i,$ $i=0,\ldots, d.$ Выберем и зафиксируем некоторое~$H \in (1/3, 1/2]$ такое, что $H<H_{\min}$. Пусть $\xi$, $\widetilde\xi$ --- случайные величины со значениями в $\mathbb{R}^n$.
\end{block}

\begin{block}{СДУ c дробными броуновскими движениями}
Рассмотрим стохастическое дифференциальное уравнение
\begin{gather}\label{chapter3_main_eq_compact}
dX_t = f(X_t)dB_t,\quad t\in[0,T],
\end{gather}
где $f=(f_0,\ldots, f_d),$ $f_i:\mathbb{R}^n \to \mathbb{R}^n,$ $i=0,\ldots,d$ --- достаточно гладкие функции с ограниченными производными. Наряду с уравнением (\ref{chapter3_main_eq_compact}) рассмотрим аналогичное уравнение с возмущенной правой частью
\begin{gather}\label{stability_intro_main_eq_1}
d\widetilde X_t = \widetilde f(\widetilde X_t)dB_t,\quad t\in[0,T],
\end{gather}
где $\widetilde f=(\widetilde f_0,\ldots, \widetilde f_d),$ $\widetilde f_i:\mathbb{R}^n \to \mathbb{R}^n,$ $i=0,\ldots,d$ --- также достаточно гладкие функции с ограниченными производными.
\end{block}

\end{frame}

\begin{frame}{Глава 3. Основные результаты}

\begin{block}{Теорема 3.3 (об асимптотических разложениях)}
Пусть $f\in C^{N+2}_b(\mathbb{R}^n, \mathbb{R}^{n \times (d+1)})$,  $g\in C^{N+3}_b(\mathbb{R}^n, \mathbb{R})$, $N\in\mathbb{N}$. Тогда для любого фиксированного $H\in(1/3, 1/2]$ такого, что $H < H_{min} = \min\limits_{i=0,\ldots d} H_i$ справедливо следующее асимптотическое разложение для функции $\textbf{P}_t g(x) = \E g(X^{x}_t)$:
\begin{equation}
\textup{\textbf{P}}_t g(x) = g(x) + \sum_{k=1}^N \sum_{I_{k}\in\{0,\ldots, d\}^{k}} t^{|H_{I_k}|}\cdot (D^{(I_{k})}_f g)(x)\E\biggl(\int_{\Delta^{k}[0,1]}dB^{(I_{k})}\biggr) + O(t^{(N+1)H}),
\end{equation}
при $t\to 0$, где $|H_{I_k}| = H_{i_1} + H_{i_2}+ \ldots + H_{i_k}$ --- сумма индексов Харста дробных броуновских движений $B^{(i_1)}, B^{(i_2)},\ldots, B^{(i_k)}$, $D^{(i)}_f = \sum_{j=1}^n f_{ji}(x)\frac{\partial}{\partial x_j}$, $i \in \{0,\ldots, d\}$, $D^{(I_k)}_f = D^{(i_1)}_f\ldots D^{(i_k)}_f$ при $I_k = (i_1,\ldots,i_k)$.
\end{block}

\begin{block}{Теорема 3.5 (аналог обратных уравнений Колмогорова)}
Предположим, что операторы $D^{(i)}_f$ и $D^{(j)}_f$ коммутируют для любых $i,j \in \{0,\ldots, d\}$. Тогда если $f\in C^\infty_b$, то функция $\varphi(t,x) = \E \left(g(X^{x}_t)\right),$
удовлетворяет дифференциальному уравнению в частных производных
\vspace{-2ex}
\begin{equation}
\frac{\partial \varphi}{\partial t} = D_f^{(0)}\varphi + \sum_{i=1}^d H_it^{2H_i - 1} (D_f^{(i)})^2 \varphi,\vspace{-2ex}
\end{equation}
с начальным условием $\varphi(0, x) = g(x).$
\end{block}

\end{frame}


\begin{frame}{Глава 3. Примеры 3.1, 3.2}

\begin{block}{Пример 3.1}
$$dX_t =  b(X_t) \,dt + \sigma(X_t)\,dB^H_t,$$
где $B^H_t$ --- одномерное дробное броуновское движение с индексом Харста $H\in(1/2,1)$, $b,\,\sigma\colon\mathbb{R}\to\mathbb{R}$ --- функции класса $C^4_b$.

Согласно теореме 3.3 асимптотические разложения примут вид:
\vspace{-2ex}
\begin{gather*}
\textbf{P}_t g(x) = g(x) + t\,b(x)Dg(x) + \frac{1}{2} t^2 \left(b(x)Db(x)Dg(x) + b^2(x) D^2g(x)\right) +\\ +\frac{1}{2} t^{2H}\left(\sigma(x)D\sigma(x)Dg(x) + \sigma^2(x) D^2g(x)\right) + O\left(t^{3H}\right).
\end{gather*}
\end{block}

\begin{block}{Пример 3.2}
$$dX_t = \sin X_t \,dt + 2H^{-1}\sin X_t \,dB^H_t,$$
где $B^H_t$ --- одномерное дробное броуновское движение с индексом Харста $H\in(1/3,1)$.

Нетрудно показать, что указанное уравнение удовлетворяет коммутативному случаю, и согласно теореме 3.5 функция $\varphi(t,x) = \E g(X^x_t)$ будет являться решением уравнения в частных производных
\vspace{-1ex}
$$\frac{\partial\varphi}{\partial t} = \left(\sin x + t^{2H-1}\sin 2x\right)\frac{\partial\varphi}{\partial x} + \frac{4t^{2H-1}}{H}\sin^2 x\, \frac{\partial^2\varphi}{\partial x^2}\vspace{-1ex}$$
с начальным условием $\varphi(0,x) = g(x)$.
\end{block}

\end{frame}


\begin{frame}{Глава 4. Основной результат}

\begin{block}{}
Будем считать функцию $f$ фиксированной, а функцию $\widetilde f$ --- изменяющейся в малой окрестности $f$ в пространстве $C^3_b(\mathbb{R}^n, \mathbb{R}^{n\times (d+1)})$.
\end{block}

\begin{block}{Теорема 4.2 (о непрерывной зависимости от начальных данных)}
Пусть $f, \widetilde f \in C^3_b(\mathbb{R}^n, \mathbb{R}^{n\times(d+1)})$, причем функция $\widetilde f$ такова, что $\|f - \widetilde f\|_{C^2_b} \le 1$. Тогда для решений $X_t$, $\widetilde X_t$ уравнений~\textup{(\ref{chapter3_main_eq_compact})}, \textup{(\ref{stability_intro_main_eq_1})} c начальными условиями $X_0=\xi$, $\widetilde X_0 = \widetilde \xi$ соответственно имеет место следующее неравенство
\begin{equation}
\mathbb{E}\bigl(\ln\|X - \widetilde X\|_H\bigr) \le C + \ln\bigl(\mathbb{E}|\xi - \widetilde\xi| + \|f-\widetilde f\|_{C^2_b}\bigr),
\end{equation}
где $C = C(H, H_1, \ldots, H_d, T, \|f\|_{C^3_b})\in\mathbb{R}$ --- константа, вообще говоря, зависящая от $H$, $H_1,\ldots, H_d$, $T$, $\|f\|_{C^3_b}$.
\end{block}
